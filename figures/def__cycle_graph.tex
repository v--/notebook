\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \n {12};
    \pgfmathsetmacro \limit {\n - 1};

    \foreach \i in { 0, ..., \limit } {
      \pgfmathtruncatemacro \angle {360/\n * (3 - \i)};
      \node[dot] (\i) at (\angle:1.5) {};
    }

    \foreach \i in { 0, ..., \limit } {
      \pgfmathtruncatemacro \j {Mod(\i + 1, \n)};
      \draw[edge] (\i) to (\j);
    }
  \end{tikzpicture}
\end{document}
