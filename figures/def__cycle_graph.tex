\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \m {12};

    \foreach \i in { 1, ..., \m } {
      \pgfmathtruncatemacro \angle {360/\m * (3 - \i)};
      \node[dot] (\i) at (\angle:1.5) {};
    }

    \pgfmathsetmacro \limit {\m - 1};
    \foreach \i in { 1, ..., \limit } {
      \pgfmathtruncatemacro \j {\i + 1};
      \draw[edge] (\i) to (\j);
    }

    \draw[edge] (\m) to (1);
  \end{tikzpicture}
\end{document}
