  \begin{tikzpicture}
    \node[dot, label=left: \( a \)] (a) at (-1,-1) {};
    \node[dot, label=right:\( b \)] (b) at (1,-1) {};
    \node[dot, label=left: \( c \)] (c) at (0,0) {};
    \node[dot, label=left: \( d \)] (d) at (-1,1) {};
    \node[dot, label=right:\( e \)] (e) at (1,1) {};

    \draw[arc] (a) to (b);
    \draw[arc] (b) to (c);
    \draw[arc] (c) to (d);
    \draw[arc] (d) to (e);
    \draw[arc] (e) to (c);
    \draw[arc] (c) to (a);
  \end{tikzpicture}
