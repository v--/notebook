\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \n {4};
    \pgfmathsetmacro \limit {\n - 1};

    \foreach \i in { 0, ..., \limit } {
      \node[dot] (\i) at (\i,0) {};
    }

    \foreach \i in { 1, ..., \limit } {
      \pgfmathtruncatemacro \j {\i - 1};
      \draw[edge] (\j) edge (\i);
    }
  \end{tikzpicture}
\end{document}
