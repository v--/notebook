\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \def \m {4};

    \foreach \i in { 1, ..., \m } {
      \node[dot] (\i) at (\i,0) {};
    }

    \pgfmathsetmacro \limit {\m - 1};
    \foreach \i in { 1, ..., \limit } {
      \pgfmathtruncatemacro \j {\i + 1};
      \draw[edge] (\i) to (\j);
    }
  \end{tikzpicture}
\end{document}
