\begin{document}
  \begin{tikzpicture}
    \node[dot, label=left:  \( b \)] (b) at (0,0) {};
    \node[dot, label=left:  \( a \)] (a) at (0,1) {};
    \node[dot, label=right: \( c \)] (c) at (1,0) {};

    \draw[arc] (a) to (b);
    \draw[arc] (a) to (c);
    \draw[arc] (b) to (c);
    \draw[arc, loop right] (a) to (a);
  \end{tikzpicture}

  \begin{tikzpicture}[scale=1.5]
    \node[dot, label=left:  \( b \)] (b) at (0,0) {};
    \node[dot, label=left:  \( a \)] (a) at (0,1) {};
    \node[dot, label=right: \( d \)] (d) at (1,0) {};
    \node[dot, label=right: \( c \)] (c) at (1,1) {};

    \draw[arc] (a) to node[left]                        {\( 3 \)} (b);
    \draw[arc] (a) to node[above]                       {\( 2 \)} (c);
    \draw[arc] (a) to node[above right, outer sep=-2pt] {\( 4 \)} (d);
    \draw[arc] (b) to node[below]                       {\( 1 \)} (d);
    \draw[arc] (c) to node[right]                       {\( 1 \)} (d);
  \end{tikzpicture}
