\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \n {6};
    \pgfmathsetmacro \limit {\n - 1};

    \foreach \i in { 0, ..., \limit } {
      \pgfmathtruncatemacro \q {floor(\i / 2)};
      \pgfmathtruncatemacro \r {Mod(\i, 2)};
      \node[dot] (\i) at (\r,\q) {};
    }

    \foreach \i in { 1, ..., \limit } {
      \pgfmathtruncatemacro \j {Mod(\i + 1, \n)};
      \draw[edge] (\i) edge (\j);
    }
  \end{tikzpicture}
