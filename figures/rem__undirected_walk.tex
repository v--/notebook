\documentclass{classes/graphs}

\NewDocumentCommand \DiamondGraphNodes {}
  {
    \node[dot, label=below:\( a \)] (a) at (0,0) {};
    \node[dot, label=left: \( b \)] (b) at (-1,1) {};
    \node[dot, label=right:\( c \)] (c) at (1,1) {};
    \node[dot, label=above:\( d \)] (d) at (0,2) {};
  }

\begin{document}
  \begin{tikzpicture}
    \DiamondGraphNodes

    \draw[arc] (a) to (b);
    \draw[arc] (b) to (c);
    \draw[arc] (c) to (a);

    \draw[arc] (d) to (b);
    \draw[arc] (c) to (d);
  \end{tikzpicture}

  \begin{tikzpicture}[arc/.append style={densely dashed}]
    \DiamondGraphNodes

    \draw[arc, solid, thick] (a) to (b);
    \draw[arc, solid, thick] (b) to (c);
    \draw[arc]               (c) to (a);

    \draw[arc]               (d) to (b);
    \draw[arc]               (c) to (d);
  \end{tikzpicture}

  \begin{tikzpicture}[edge/.append style={densely dashed}, arc/.append style={thick}]
    \DiamondGraphNodes

    \draw[arc]  (a) to (b);
    \draw[arc]  (b) to (c);
    \draw[edge] (c) to (a);

    \draw[edge] (d) to (b);
    \draw[edge] (c) to (d);
  \end{tikzpicture}

  \begin{tikzpicture}[arc/.append style={densely dashed}, scale=1.5]
    \DiamondGraphNodes;

    \draw[arc, solid, thick] (a) to               (b);
    \draw[arc]               (a) to[bend right]   (c);
    \draw[arc]               (b) to[bend right]   (a);
    \draw[arc, solid, thick] (b) to[bend left=15] (c);
    \draw[arc]               (b) to[bend left]    (d);
    \draw[arc]               (c) to[bend left=15] (b);
    \draw[arc]               (c) to               (a);
    \draw[arc]               (c) to               (d);
    \draw[arc]               (d) to               (b);
    \draw[arc]               (d) to[bend left]    (c);
  \end{tikzpicture}

  \begin{tikzpicture}[arc/.append style={densely dashed}, scale=1.5]
    \DiamondGraphNodes;

    \draw[arc]               (a) to               (b);
    \draw[arc]               (a) to[bend right]   (c);
    \draw[arc, solid, thick] (b) to[bend right]   (a);
    \draw[arc]               (b) to[bend left=15] (c);
    \draw[arc]               (b) to[bend left]    (d);
    \draw[arc, solid, thick] (c) to[bend left=15] (b);
    \draw[arc]               (c) to               (a);
    \draw[arc]               (c) to               (d);
    \draw[arc]               (d) to               (b);
    \draw[arc]               (d) to[bend left]    (c);
  \end{tikzpicture}
\end{document}
