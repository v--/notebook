      \draw[edge] (0) to (\i);
    }
  \end{tikzpicture}

  \begin{tikzpicture}
    \foreach \i in { 1, 2, 3 } {

    }

    \foreach \i in { 1, 2, 3 } {
      \foreach \j in { 4, 5, 6 } {
        \draw[edge] (\i) to (\j);

    }
