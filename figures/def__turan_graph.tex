\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \m {5};

    \foreach \i in { 1, ..., \m } {
      % -7/4 is a correction so that the graph would "stand" upright.
      \node[dot] (\i) at ({(\i - 7/4) * 360/\m}: 1) {};
    }

    \foreach \i in { 1, ..., \m } {
      \foreach \j in { 1, ..., \i } {
        \draw[edge] (\i) edge (\j);
      }
    }
  \end{tikzpicture}
