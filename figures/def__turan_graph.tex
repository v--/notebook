\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}
    \pgfmathsetmacro \n {5};
    \pgfmathsetmacro \limit {\n - 1};

    \foreach \i in { 0, ..., \limit } {
      % -3/4 is a correction so that the graph would "stand" upright.
      \node[dot] (\i) at ({(\i - 3/4) * 360/\n}: 1) {};
    }

    \foreach \i in { 0, ..., \limit } {
      \foreach \j in { 0, ..., \i } {
        \draw[edge] (\i) edge (\j);
      }
    }
  \end{tikzpicture}

  \begin{tikzpicture}

      % -3/4 is a correction so that the graph would "stand" upright.
      \node[dot] (\i) at ({(\i - 3/4) * 360/\n}: 1) {};
    }

    \draw[edge] (0) edge (2);
    \draw[edge] (0) edge (3);
    \draw[edge] (0) edge (4);

    \draw[edge] (1) edge (2);
    \draw[edge] (1) edge (3);
    \draw[edge] (1) edge (4);

    \draw[edge] (2) edge (3);
    \draw[edge] (2) edge (4);
  \end{tikzpicture}
\end{document}
