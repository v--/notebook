\documentclass{classes/graphs}

\begin{document}
  \begin{tikzpicture}[scale=1.5]
    \pgfmathsetmacro \n {5};
    \pgfmathsetmacro \limit {\n - 1};

    \foreach \i in { 0, ..., \limit } {
      % We apply a rotation so that the edge coloring would look more natural
      \node[dot] (\i) at ({(\i + 1/4) * 360/\n}: 1) {};
    }

    \foreach \i in { 1, ..., \limit } {
      \foreach \j in { 0, ..., \i } {
        \draw[edge, lightgray] (\i) edge (\j);
      }
    }

    \foreach \i in { 0, ..., \limit } {
      \pgfmathtruncatemacro \j {Mod(\i + 1, \n)};
      \draw[edge] (\i) edge (\j);
      \node[dot] at (\i) {};
    }
  \end{tikzpicture}
\end{document}
