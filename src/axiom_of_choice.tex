\subsection{Axiom of choice}\label{subsec:axiom_of_choice}

\begin{remark}\label{rem:aoc}
  \Fullref{thm:axiom_of_choice_equivalences} famously leads to counterintuitive results, which is a reason why it is frowned upon. It is insanely useful in the form of its many equivalent statements (see \fullref{def:zfc/A8}).
\end{remark}

\begin{definition}\label{def:choice_function}
  Let \( X \) be a nonempty set. A \term{choice function} is a function of type \( f: \pow(X) \to X \) such that, for all subsets \( A \subseteq X \),
  \begin{equation*}
    f(A) \in A.
  \end{equation*}

  That is, the function \( f \) \enquote{chooses} an element out of any subset of \( X \).
\end{definition}

\begin{theorem}[Equivalent to AOC theorems]\label{thm:axiom_of_choice_equivalences}
  The following are equivalent:

  \begin{thmenum}
    \thmitem{thm:axiom_of_choice_equivalences/transversal}\mcite[thm. 6M(4)]{Enderton1977Sets}Every hypergraph has a minimal transversal (compare to \fullref{thm:finite_hypergraphs_have_minimal_transversal} that only mentions finite sets).

    \thmitem{thm:axiom_of_choice_equivalences/product}\mcite[thm. 6M(2)]{Enderton1977Sets}The \hyperref[def:cartesian_product]{Cartesian product} of a family of nonempty sets is nonempty.

    \medskip

    \thmitem{thm:axiom_of_choice_equivalences/choice}\mcite[thm. 6M(3)]{Enderton1977Sets}For any nonempty set \( X \) there exists a \hyperref[def:choice_function]{choice function}.

    \medskip

    \thmitem{thm:axiom_of_choice_equivalences/function}\mcite[thm. 6M(1)]{Enderton1977Sets}Every \hyperref[def:function/multivalued]{multivalued function} has a \hyperref[def:function/selection]{selection}.

    \thmitem{thm:axiom_of_choice_equivalences/cardinals} \Fullref{thm:cardinal_trichotomy}.

    \thmitem{thm:axiom_of_choice_equivalences/zorn} \Fullref{thm:zorns_lemma}.

    \thmitem{thm:axiom_of_choice_equivalences/well_ordering_principle} \Fullref{thm:well_ordering_principle}.

    \thmitem{thm:axiom_of_choice_equivalences/vector_space_bases} \Fullref{thm:all_vector_spaces_are_free_left_modules}. Compare this to finite-dimensional vector spaces of order \( n \) over \( \BbbK \), which are all isomorphic to \( \BbbK^n \).

    \thmitem{thm:axiom_of_choice_equivalences/tychonoff} \Fullref{thm:tychonoffs_product_theorem}.

    \thmitem{thm:axiom_of_choice_equivalences/krull} \Fullref{thm:krulls_theorem}.
  \end{thmenum}
\end{theorem}

\begin{theorem}[Diaconescu-Goodman-Myhill theorem]\label{diaconescu_goodman_myhill_theorem}\mcite[corr. 2]{Diaconescu1975}
  In \hyperref[def:set]{ZFC}, the \hyperref[def:zfc/A9]{axiom of choice} entails the law of the excluded middle \eqref{eq:thm:minimal_propositional_negation_laws/lem}.
\end{theorem}
