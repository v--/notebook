\section{Logic}\label{sec:logic}
\subsection{Languages}\label{subsec:languages}

\begin{remark}\label{rem:language_definitions_using_sets}
  Languages are used to define formulas for expressing the \hyperref[def:set_zfc]{axioms of set theory}. Here, sets are used to formally define languages. Finding a way out of this vicious cycle is left to logicians.
\end{remark}

\begin{definition}\label{def:language}
  Fix a nonempty set \( \mscrA \).

  \begin{DefEnum}
    \ILabel{def:language/alphabet} We call \( \mscrA \) an \Def{alphabet}.

    \ILabel{def:language/symbol} We call each element of \( \mscrA \) a \Def{symbol}.

    \ILabel{def:language/word} A \Def{word} over \( \mscrA \) is a \hyperref[def:cartesian_product]{tuple} of symbols. If \( (a, b, c) \) is a word, for convenience we write it as the string \( abc \). This is the reason words are also referred to as \Def{strings}. This notation only makes sense if each symbol of the language is actually represented by a symbol.

    \ILabel{def:language/empty_word} We denote the empty word by \( \varepsilon \).

    \ILabel{def:language/word_length} The \Def{length} \( \Len(w) \) of a word \( w \) is the number of elements of the tuple \( w \).

    \ILabel{def:language/concatenation} The \Def{concatenation} of the words \( v = (v_1, \ldots, v_n) \) and \( w = (w_1, \ldots, w_m) \) is the word
    \begin{equation*}
      vw \coloneqq (v_1, \ldots, v_n, w_1, \ldots, w_m).
    \end{equation*}

    We abbreviate \( \underbrace{w w \ldots w}_{k \T{times}} \) as \( w^k \). This is only a notation. We do not distinguish, formally, between the words \( aaabbaa \) and \( a^3 b^2 a^2 \).

    \ILabel{def:language/reverse} The \Def{reverse word} of \( w = (w_1, \ldots, w_n) \) is
    \begin{equation*}
      \Op{Rev}(w) \coloneqq (w_n, \ldots, w_1).
    \end{equation*}

    \ILabel{def:language/prefix} The word \( p = (p_1, \ldots, p_m) \) is a \Def{prefix} of \( w = (w_1, \ldots, w_n) \) if
    \begin{equation*}
      w = (\underbrace{p_1, \ldots, p_m}_p, w_{m+1}, \ldots, w_n).
    \end{equation*}

    \ILabel{def:language/suffix} The word \( s \) is a \Def{suffix} of \( w \) if \( \Op{Rev}(s) \) is a prefix of \( \Op{Rev}(w) \).

    \ILabel{def:language/subword} The word \( v \) is a \Def{subword} of \( w \) if there exists a prefix \( p \) and a suffix \( s \) of \( v \) such that
    \begin{equation*}
      w = pvs.
    \end{equation*}

    \ILabel{def:language/kleene_star} The \Def{Kleene star} \( \mscrA^{\ast} \) of \( \mscrA \) is the set of all (finite) words over \( \mscrA \).

    \ILabel{def:language/language} A \Def{language} over \( \mscrA \) is any subset of \( \mscrA^{\ast} \). Note that, in some contexts like \hyperref[subsec:language_of_propositional_logic]{propositional logic} or \hyperref[subsec:language_of_first_order_logic]{first-order logic}, the term \enquote{language} may refer to the alphabet itself (see \fullref{rem:propositional_language_is_alphabet}).
  \end{DefEnum}
\end{definition}

\begin{proposition}\label{thm:kleene_star_is_monoid}
  For any alphabet \( \mscrA \), the Kleene star \( \mscrA^{\ast} \) is a \hyperref[def:unital_magma/associative]{monoid} under concatenation.
\end{proposition}
\begin{proof}
  Concatenation is clearly associative and the empty word \( \varepsilon \) is a \hyperref[def:magma_identity]{two-sided identity} under concatenation.
\end{proof}
