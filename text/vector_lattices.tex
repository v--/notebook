\subsection{Vector lattices}\label{subsec:vector_lattices}

\begin{definition}\label{def:ordered_vector_space}\mcite[3]{Peressini1967}
  An \term{ordered vector space} is a real \hyperref[def:vector_space]{vector space} \( X \) endowed with a \hyperref[def:partially_ordered_set]{partial order} \( \leq \) such that \( (X, +, \leq) \) is an \hyperref[def:ordered_magma]{ordered group} and, for any positive number \( \alpha \), \( x \leq y \) implies \( \alpha x \leq \alpha y \).

  We define \( \abs{x} \coloneqq x \vee -x \).

  If \( (X, \leq) \) is a \hyperref[def:lattice]{lattice}, we call the space a \term{vector lattice}.
\end{definition}

\begin{definition}\label{def:solid_set}\mcite[def. 4.1]{Peressini1967}
  We say that a set \( A \) in some \hyperref[def:ordered_vector_space]{vector lattice} is \term{solid} if \( x \in A \) and \( \norm{y} \leq \norm{x} \) implies that \( y \in A \).
\end{definition}

\begin{definition}\label{def:ordered_topological_vector_space}\mcite[62]{Peressini1967}
  An \term{ordered topological vector space} is a real \hyperref[def:topological_vector_space]{topological vector space} that is also an \hyperref[def:ordered_vector_space]{ordered vector space}.
\end{definition}

\begin{definition}\label{def:topological_vector_lattice}\mcite[def. 4.6]{Peressini1967}
  A \term{topological vector lattice} is a real \hyperref[def:topological_vector_space]{topological vector space} that is also an \hyperref[def:ordered_vector_space]{vector lattice} such that every point has a \hyperref[def:topological_local_base]{local base} consisting of \hyperref[def:solid_set]{solid sets}.
\end{definition}

\begin{definition}\label{def:normed_vector_lattice}
  We say that a \hyperref[def:ordered_vector_space]{vector lattice} is \term{normed} by \( \norm{\anon} \) if \( \abs{x} \leq \abs{y} \) implies \( \norm{x} \leq \norm{y} \).

  If the space is complete with respect to this norm, we say that it is a \term{Banach lattice}.
\end{definition}
