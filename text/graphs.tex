\begin{definition}\label{def:directed_multigraph}\mcite[def. 1.1.1; def. 1.1.2]{Knauer2019AlgebraicGraphTheory}
  A \term[bg=ориентиран (\cite[6]{Мирчев2001Графи}) мултиграф (\cite[7]{Мирчев2001Графи}), ru=ориентированый мультиграф (\cite[16]{ЕмеличевИПр1990ТеорияГрафов})]{directed multigraph} is a quadruple \( G = (V, A, h, t) \) consisting of:
  \begin{thmenum}[series=def:directed_multigraph]
    \thmitem{def:directed_multigraph/vertices} A set \( V \), whose elements we call \term[bg=върхове (\cite[6]{Мирчев2001Графи}), ru=вершины (\cite[279]{ЕмеличевИПр1990ТеорияГрафов})]{vertices}.

    \thmitem{def:directed_multigraph/arcs} A disjoint from \( V \) set \( A \), whose elements we call \term[bg=дъги (\cite[6]{Мирчев2001Графи}), ru=дуги (\cite[279]{ЕмеличевИПр1990ТеорияГрафов})]{arcs}.

    \thmitem{def:directed_multigraph/head} A function \( h: A \to V \), giving the \term[en=head (\cite[191]{Erickson2019Algorithms})]{head} or \term[bg=начален връх (\cite[7]{Мирчев2001Графи}), ru=начало (\cite[279]{ЕмеличевИПр1990ТеорияГрафов}), en=initial vertex (\cite[28]{Diestel2017GraphTheory})]{initial endpoint} of an arc.

    \thmitem{def:directed_multigraph/tail} A function \( t: A \to V \), giving the \term[en=tail (\cite[191]{Erickson2019Algorithms})]{tail} or \term[bg=краен връх (\cite[7]{Мирчев2001Графи}), ru=конец (\cite[279]{ЕмеличевИПр1990ТеорияГрафов}), en=terminal vertex (\cite[28]{Diestel2017GraphTheory})]{terminal endpoint} of an arc.
  \end{thmenum}

  \begin{figure}
    \begin{equation}\label{eq:fig:def:directed_multigraph}
      \begin{aligned}
        \includegraphics[page=1]{output/def__directed_multigraph}
      \end{aligned}
    \end{equation}
    \caption{A \hyperref[def:directed_multigraph]{directed multigraph} with a pair of parallel arcs, a pair of oppositely directed arcs and a loop. Removing the dashed arcs \( \widehat{e}_2 \), \( \widehat{e}_6 \) and \( e_8 \) makes it a \hyperref[def:directed_graph]{simple directed graph}.}\label{fig:def:directed_multigraph}
  \end{figure}

  For the arcs of the graph \( G = (V, A, h, t) \), we introduce several auxiliary notions:
  \begin{thmenum}[resume=def:directed_multigraph]
    \thmitem{def:directed_multigraph/loop} We call the arc \( e \) a \term[bg=примка (\cite[7]{Мирчев2001Графи}), ru=петля (\cite[279]{ЕмеличевИПр1990ТеорияГрафов})]{loop} if \( h(e) = t(e) \).

    \thmitem{def:directed_multigraph/parallel_arcs}\mcite[28]{Diestel2017GraphTheory} We say that two distinct arcs \( e \) and \( f \) are \term[bg=паралелни (ребра) (\cite[7]{Мирчев2001Графи}), ru=параллельные (рёбра) (\cite[279]{ЕмеличевИПр1990ТеорияГрафов})]{parallel} if \( h(e) = h(f) \) and \( t(e) = t(f) \).

    \thmitem{def:directed_multigraph/opposite_arcs}\mimprovised We say that two distinct arcs \( e \) and \( f \) are \term{opposite} if \( h(e) = t(f) \) and \( t(e) = h(f) \).
  \end{thmenum}

  Directed multigraphs have the following metamathematical properties:
  \begin{thmenum}[resume=def:directed_multigraph]
    \thmitem{def:directed_multigraph/morphism}\mcite[def. 1.4.1]{Knauer2019AlgebraicGraphTheory} A \term{homomorphism} between the directed multigraphs \( G \) and \( H \) is a pair of functions
    \begin{align*}
      \varphi_V &: V_G \to V_H, \\
      \varphi_A &: A_G \to A_H,
    \end{align*}
    such that
    \begin{align}
      h_H \bincirc \varphi_A &= \varphi_V \bincirc h_G, \label{eq:def:directed_multigraph/morphism/head} \\
      t_H \bincirc \varphi_A &= \varphi_V \bincirc t_G. \label{eq:def:directed_multigraph/morphism/tail}
    \end{align}

    The morphism-related notions from \fullref{sec:categories} apply here. Isomorphisms in particular are discussed in \cref{thm:graph_isomorphisms/directed_multigraph}.

    \thmitem{def:directed_multigraph/category}\mcite[48]{MacLane1998CategoryTheory} For a fixed Grothendieck universe \( \mscrU \), we can define the \hyperref[def:category]{category} of \( \mscrU \)-small directed multigraphs as follows:

    \begin{thmenum}
      \thmitem{def:directed_multigraph/category/objects} The \hyperref[def:category/objects]{set of objects} is the family of directed multigraphs, where both the set of vertices and the set of arcs are \( \mscrU \)-small.

      \thmitem{def:directed_multigraph/category/morphisms} The \hyperref[def:category/morphisms]{morphisms} between two directed multigraphs are their homomorphisms as defined above.

      \thmitem{def:directed_multigraph/category/composition} The \hyperref[def:category/composition]{composition of the morphisms} \( (\varphi_V, \varphi_A): G \to H \) and \( (\psi_V, \psi_A): H \to K \) is their componentwise function composition
      \begin{equation*}
        (\psi_V \bincirc \varphi_V, \psi_A \bincirc \varphi_A): G \to K.
      \end{equation*}

      \thmitem{def:directed_multigraph/category/identity} The \hyperref[def:category/identity]{identity morphism} on the directed multigraph \( G \) is \( (\id_{V_G}, \id_{A_G}) \).
    \end{thmenum}

    \thmitem{def:directed_multigraph/subobject}\mimprovised We say that \( H \) is a \term{subgraph} of \( G \) if \( V_G \subseteq V_H \), \( A_G \subseteq A_H \) and \( h_H \) and \( t_H \) are restrictions of \( h_G \) and \( t_G \) to \( A_H \).

    As we will show in \cref{thm:subgraphs}, subgraphs correspond to \hyperref[def:categorical_subobject]{categorical subobjects}.

    \thmitem{def:directed_multigraph/opposite}\mimprovised We define the \term{opposite} of \( G = (V, A, t, h) \) by reversing the directions of all arcs:
    \begin{equation*}
      G^\oppos \coloneqq (V, A, h, t).
    \end{equation*}

    This is an \hyperref[con:opposite_object]{opposite object functor} in categories of directed multigraphs.
  \end{thmenum}
\end{definition}
\begin{comments}
  \item It is possible to formalize directed multigraphs in \hyperref[def:many_sorted_fol]{two-sorted first-order logic} --- see \cref{rem:logical_theory_of_graphs/directed_multigraphs}.

  \item \Cref{fig:def:directed_multigraph} illustrates this definition. The figure is not merely schematic --- it is a \hyperref[def:graph_geometric_realization/embedding]{graph embedding}.

  \item \Textcite[675]{Rosen2019DiscreteMathematics} defines \enquote{directed multigraphs} similarly to us, but prefers to treat the edge set as a multiset rather than use head and tail functions.

  \textcite[def. 1.1.1]{Knauer2019AlgebraicGraphTheory}, \textcite[10]{MacLane1998CategoryTheory} and \textcite[28]{Diestel2017GraphTheory} provide definitions similar to ours, but call them \enquote{directed graphs}. Knauer later distinguishes between simple and multigraphs, while Diestel uses \enquote{oriented graph} for what we call \enquote{simple directed graph}. \Textcite[8]{Bollobás1998ModernGraphTheory} briefly mentions directed multigraphs and calls them as such.

  Diestel and Bollob\'as reserve the term \enquote{multigraph} for \hyperref[def:hypergraph/multigraph]{undirected multigraphs}. The latter also implicitly restricts graphs to finite order. Their books use \hyperref[def:undirected_graph]{simple undirected graphs} almost exclusively.

  \Textcite[1]{König1986Graphentheorie}, in the first book on graph theory, gives a vague definition of a graph that roughly coincides with our notion of directed multigraph, with infinite graphs explicitly considered.
\end{comments}

\begin{remark}\label{rem:digraph}
  The term \enquote{digraph} is used as a shorthand for \enquote{directed graph}, for example in
  \cite[10]{Harary1969GraphTheory},
  \cite[def. 1.1.1]{Knauer2019AlgebraicGraphTheory},
  \cite[def. 10.1.2]{Rosen2019DiscreteMathematics},
  \cite[571]{Stanley2012EnumerativeCombinatoricsVol1} and
  \cite[372]{Knuth1997ArtVol1}.

  \Textcite[28]{Diestel2017GraphTheory} also uses this terminology, and in his usage \enquote{graph} exclusively refers to \enquote{undirected graph}.

  Similarly, in Russian, \enquote{орграф} is used as a shorthand for \enquote{ориентированный граф}, for example in
  \cite[\S 7.1.5]{Новиков2013ДискретнаяМатематика} and
  \cite[279]{ЕмеличевИПр1990ТеорияГрафов}.

  We avoid this term because we prefer being more explicit.
\end{remark}

\begin{definition}\label{def:directed_graph}\mcite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory}
  A \term{simple directed graph} is a pair \( G = (V, A) \) consisting of the following:
  \begin{thmenum}[series=def:directed_graph]
    \thmitem{def:directed_graph/vertices} A set \( V \), whose elements we call \term{vertices}.

    \thmitem{def:directed_graph/arcs} A disjoint from \( V \) set \( A \) of \hyperref[def:ordered_tuple]{ordered pairs} of \emph{distinct} vertices. As in the case of directed multigraphs, we call the elements of \( A \) \term{arcs}. If we allow vertices whose endpoints coincide, we will say that \( G \) is a \enquote{simple directed graph, with loops allowed}.
  \end{thmenum}

  Simple directed graphs have the following metamathematical properties:
  \begin{thmenum}[resume=def:directed_graph]
    \thmitem{def:directed_graph/morphism}\mimprovised A \term{homomorphism} between the directed graphs \( G \) and \( H \) is a function \( \varphi: V_G \to V_H \) such that
    \begin{equation*}
      (u, v) \in A_G \T{implies} (f(u), f(v)) \in A_H
    \end{equation*}

    The morphism-related notions from \fullref{sec:categories} apply here. Isomorphisms in particular are discussed in \cref{thm:graph_isomorphisms/simple_directed}.

    \thmitem{def:directed_graph/subobject}\mimprovised We say that \( H \) is a \term{subgraph} of \( G \) if \( V_G \subseteq V_H \) and \( A_G \subseteq A_H \).

    As we will show in \cref{thm:subgraphs}, subgraphs correspond to \hyperref[def:categorical_subobject]{categorical subobjects}.

    \thmitem{def:directed_graph/category}\mimprovised For a fixed Grothendieck universe \( \mscrU \), we can define \hyperref[def:category]{categories} of \( \mscrU \)-small simple directed graphs, with and without loops, along with their homomorphisms.

    \thmitem{def:directed_graph/opposite}\mimprovised Similarly to \hyperref[def:directed_multigraph/opposite_arcs]{opposite directed multigraphs}, given \( G = (V, A) \), we define its \term{opposite} \( G^\oppos \) as \( (V, A^\oppos) \), where
    \begin{equation*}
      A^\oppos \coloneqq \set{ (v, u) \given (u, v) \in A }.
    \end{equation*}
  \end{thmenum}
\end{definition}
\begin{comments}
  \item We can regard directed graphs as directed multigraphs without loops and parallel arcs. This is made precise via the inclusion functor \hyperref[def:graph_functors/directed_inclusion]{\( I_D \)}.

  \item It is possible to formalize directed graphs in \hyperref[def:first_order_logic]{one-sorted first-order logic} --- see \cref{rem:logical_theory_of_graphs/directed_graphs}.
\end{comments}

\begin{proposition}\label{def:directed_graph_induced_by_relation}
  Every \hyperref[def:binary_relation]{binary relation} \( R \) on the set \( A \) induces the \hyperref[def:directed_graph]{directed graph} \( G = (A, R) \), in which there is an arc from \( x \) to \( y \) and only if \( x R y \).
\end{proposition}
\begin{proof}
  Trivial.
\end{proof}

\begin{definition}\label{def:hypergraph}\mcite[185]{DörflerWaller1980Hypergraphs}
  A \term[ru=гиперграф (\cite[298]{ЕмеличевИПр1990ТеорияГрафов})]{hypergraph} is a triple \( G = (V, E, \mscrE) \) consisting of the following:
  \begin{thmenum}[series=def:hypergraph]
    \thmitem{def:hypergraph/vertices} A set \( V \), whose elements we call \term[ru=вершины (\cite[298]{ЕмеличевИПр1990ТеорияГрафов})]{vertices}.

    \thmitem{def:hypergraph/edges} A disjoint from \( V \) set \( E \), whose elements we call \term[ru=рёбра (\cite[298]{ЕмеличевИПр1990ТеорияГрафов})]{edges}.

    \thmitem{def:hypergraph/endpoints} A map \( \mscrE: E \to \pow(V) \setminus \set{ \varnothing } \), providing a nonempty set of \term{endpoints} for every edge.
  \end{thmenum}

  \begin{figure}
    \begin{equation}\label{eq:fig:def:hypergraph}
      \begin{aligned}
        \includegraphics[page=1]{output/def__hypergraph}
      \end{aligned}
    \end{equation}
    \caption{A hypergraph, which becomes an undirected multigraph after removing \( e_9 \) and a simple graph after further removing \( \widehat{e}_2 \), \( \widehat{e}_6 \) and \( e_8 \).}\label{fig:def:hypergraph}
  \end{figure}

  For the arcs of the hypergraph \( G = (V, E, \mscrE) \), we introduce several auxiliary notions:
  \begin{thmenum}[resume=def:hypergraph]
    \thmitem{def:hypergraph/loop} We call the edge \( e \) a \term{loop} if \( \mscrE(e) \) is a singleton set.

    \thmitem{def:hypergraph/hyperedge} We say that \( e \) is a \term[ru=гипердуги (\cite[\S 7.1.5]{Новиков2013ДискретнаяМатематика}), en=hyperedge (\cite[7]{Bollobás1998ModernGraphTheory})]{hyperedge} if it has more than two endpoints.

    \thmitem{def:hypergraph/parallel}\mimprovised We call the (hyper)edges \( e \) and \( f \) \term{parallel} if \( \mscrE(e) = \mscrE(f) \).
  \end{thmenum}

  Hypergraphs have the following metamathematical properties:
  \begin{thmenum}[resume=def:hypergraph]
    \thmitem{def:hypergraph/morphism}\mcite[186]{DörflerWaller1980Hypergraphs} A \term{homomorphism} between the hypergraphs \( G \) and \( H \) is a pair of functions
    \begin{align*}
      &\varphi_V: V_G \to V_H, \\
      &\varphi_E: E_G \to E_H,
    \end{align*}
    such that, for each edge \( e \in E_G \),
    \begin{equation}\label{eq:def:hypergraph/morphism}
      \mscrE_H(\varphi_E(e)) = \set[\big]{ \varphi(v) \given* v \in \mscrE_G(e) }.
    \end{equation}

    The morphism-related notions from \fullref{sec:categories} apply here. Isomorphisms in particular are discussed in \cref{thm:graph_isomorphisms/hypergraphs}.

    \thmitem{def:hypergraph/category}\mcite[186]{DörflerWaller1980Hypergraphs} For a fixed Grothendieck universe \( \mscrU \), we can define the \hyperref[def:category]{category} of \( \mscrU \)-small hypergraphs and their homomorphisms with componentwise composition, akin to how we defined the category of \( \mscrU \)-small directed multigraphs in \cref{def:directed_multigraph}.

    \thmitem{def:hypergraph/subobject}\mimprovised We say that \( H \) is a \term{sub(hyper)graph} of \( G \) if \( V_G \subseteq V_H \), \( E_G \subseteq E_H \) and \( \mscrE_H = \mscrE_G\restr_{E_H} \).

    As we will show in \cref{thm:subgraphs}, sub(hyper)graphs correspond to \hyperref[def:categorical_subobject]{categorical subobjects}.

    \thmitem{def:hypergraph/multigraph} We say that \( G \) is an \term{undirected multigraph} if it has no hyperedges, i.e. if all edges have either zero or one vertices.

    \thmitem{def:hypergraph/uniform}\mcite[191]{DörflerWaller1980Hypergraphs} If all edges have \( \kappa \) vertices, we say that the hypergraph is \term[ru=\( \kappa \)-однородный]{\( \kappa \)-uniform}.
  \end{thmenum}
\end{definition}
\begin{comments}
  \item We show in \cref{rem:logical_theory_of_graphs/hypergraphs} that hypergraphs do not allow a straightforward \hyperref[def:logical_theory]{logical theory}, unlike directed (multi)graphs, or even undirected simple graphs.

  \item We base our exposition of hypergraphs on \cite{DörflerWaller1980Hypergraphs}. The term \enquote{undirected multigraphs} is not as established.
  \begin{itemize}
    \item \Textcite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory} does not explicitly mention undirected multigraphs, but their existence is implied.

    \item Without the prefix \enquote{undirected}, multigraphs are defined by
    \textcite[10]{Harary1969GraphTheory},
    \textcite[28]{Diestel2017GraphTheory},
    \textcite[6]{Bollobás1998ModernGraphTheory} and
    \textcite[674]{Rosen2019DiscreteMathematics}.

    Harary disallows loops, instead referring to multigraphs with loops as \enquote{pseudographs}. Bollob\'as explicitly defines directed multigraphs, while Diestel uses the term \enquote{directed graph} for what we call a \hyperref[def:directed_multigraph]{directed multigraph} and \enquote{oriented graph} for what we call \enquote{simple directed graph}.

    \item \Textcite[3]{GondranMinoux1984GraphsAndAlgorithms} define multigraphs without specifying whether they are directed or not, but later implicitly assume that they are undirected.
  \end{itemize}
\end{comments}

\begin{definition}\label{def:undirected_graph}\mcite[2; 3]{Diestel2017GraphTheory}
  A \term{simple undirected graph} is a pair \( G = (V, E) \) consisting of the following:
  \begin{thmenum}[series=def:undirected_graph]
    \thmitem{def:undirected_graph/vertices} A set \( V \), whose elements we call \term{vertices}.
    \thmitem{def:undirected_graph/edges} A disjoint from \( V \) set \( E \) of unordered pairs of \emph{distinct} vertices, i.e. sets of the form \( \set{ u, v } \), whose elements we call \term{edges}. If we allow vertices whose endpoints coincide, we will say that \( G \) is a \enquote{simple undirected graph, with loops allowed}.
  \end{thmenum}

  We will use the following basic terminology:
  \begin{thmenum}[resume=def:undirected_graph]
    \thmitem{def:undirected_graph/morphism} A \term{homomorphism} between the undirected graphs \( G \) and \( H \) is a function \( \varphi: V_G \to V_H \) such that
    \begin{equation*}
      \set{ u, v } \in E_G \T{implies} \set{ f(u), f(v) } \in E_H.
    \end{equation*}

    The morphism-related notions from \fullref{sec:categories} apply here. Isomorphisms in particular are discussed in \cref{thm:graph_isomorphisms/simple_undirected}.

    \thmitem{def:undirected_graph/category}\mcite[example 3.1.12]{Knauer2019AlgebraicGraphTheory} For a fixed Grothendieck universe \( \mscrU \), we can define \hyperref[def:category]{categories} of \( \mscrU \)-small simple undirected graphs, with and without loops, along with their homomorphisms.

    We list several properties:
    \begin{itemize}
      \item The forgetful functor sending a simple undirected graph to its underlying vertices has both a left adjoint --- the \hyperref[def:edgeless_graph]{edgeless graphs} characterized by \fullref{thm:edgeless_graph_universal_property} --- and a right adjoint --- the \hyperref[def:complete_graph]{complete graph}, characterized by \fullref{thm:complete_graph_universal_property}.

      \item Based on the above, \cref{thm:concrete_category_function_invertibility} implies that the \hyperref[def:morphism_invertibility/left_cancellative]{monomorphisms} are precisely the injective homomorphisms, while the \hyperref[def:morphism_invertibility/right_cancellative]{epimorphisms} are precisely the surjective homomorphisms.
    \end{itemize}

    \thmitem{def:undirected_graph/subobject}\mcite[4]{Diestel2017GraphTheory} We say that \( H \) is a \term{subgraph} of \( G \) if \( V_G \subseteq V_H \) and \( E_G \subseteq E_H \).
  \end{thmenum}
\end{definition}
\begin{comments}
  \item We can regard undirected graphs as undirected multigraphs without loops and parallel arcs. This is made precise via the \hyperref[con:inclusion_and_projection]{inclusion} functor \hyperref[def:graph_functors/undirected_inclusion]{\( I_U \)}.

  Conversely, although simple undirected graphs are much less general, \cref{thm:hypergraph_representation} allows us to regard hypergraphs as \hyperref[def:graph_coloring/proper]{properly \( 2 \)-colored} simple graphs.

  \item It is possible to formalize simple undirected graphs in \hyperref[def:first_order_logic]{one-sorted first-order logic} as special cases of simple directed graphs --- see \cref{rem:logical_theory_of_graphs/undirected_graphs}.

  \item A similar definition is given in
  \cite[9]{Harary1969GraphTheory},
  \cite[2]{Diestel2017GraphTheory} and
  \cite[1]{Bollobás1998ModernGraphTheory}.

  A similar definition not forbidding loops is given in
  \cite[def. 10.1.1]{Rosen2019DiscreteMathematics} and
  \cite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory}.

  \item \Textcite[def. 1.4.8]{Knauer2019AlgebraicGraphTheory} define subgraphs more generally as graphs which can be embedded via injective graph homomorphisms. We prefer the vertices and edges of subgraphs to be subsets.
\end{comments}

\begin{remark}\label{rem:logical_theory_of_graphs}
  There are multiple approaches for describing graphs using the formalisms of \fullref{ch:mathematical_logic}. We will present the ones that are most obvious, highlighting some issues along the way.

  \begin{thmenum}
    \thmitem{rem:logical_theory_of_graphs/directed_multigraphs} A \hyperref[def:directed_multigraph]{directed multigraph} can be described in \hyperref[def:many_sorted_fol]{many-sorted first-order logic} using the sorts \( \syn\kappa \) (from \enquote{\textgr{κορυφή}}) for vertices and \( \syn\alpha \) (from \enquote{\textgr{άκρη}}) for edges.

    We need two logical constants --- \( \synh: \syn\kappa \synimplies \syn\alpha \) and \( \synt: \syn\kappa \synimplies \syn\alpha \).

    As we will show in \cref{ex:fol_to_hol_homomorphism/directed_multigraphs}, a \hyperref[def:hol_homomorphism]{higher-order homomorphism} \( \varphi: G \to H \) must satisfy
    \begin{equation*}
      h_H(\varphi_{\syn\kappa}(e)) = \varphi_{\syn\kappa}(h_G(e)),
    \end{equation*}
    and similarly for tails. This coincides with our homomorphism definition from \cref{def:directed_multigraph/morphism}.

    \thmitem{rem:logical_theory_of_graphs/hypergraphs} The situation for \hyperref[def:hypergraph]{hypergraphs} is less fortunate. We can try describing it using \hyperref[def:second_order_logic]{second-order logic} using only one sort, by using \( \syn\kappa \) for vertices and \( \syn\kappa \synimplies \syn\omicron \) for hyperedges.

    This approach unfortunately gives us little insight into the models of this theory because there are no signature symbols and no axioms. One solution is to add a mock symbol,
    \begin{equation*}
      \synin: \syn\kappa \synimplies (\syn\kappa \synimplies \syn\omicron) \synimplies \syn\omicron,
    \end{equation*}
    that is to be interpreted as in \( G = (V, E, \mscrE) \) by checking whether \( v \) is in \( \mscrE(e) \).

    If we are to use the \hyperref[def:hol_expression]{higher-order expression} \enquote{\( v \synin e \)} where we would otherwise use \enquote{\( ev \)}, relying second-order logic becomes pointless, and we can instead rely on many-sorted first-order logic with two sorts --- \( \syn\kappa \) for vertices, and \( \syn\alpha \) for nodes (form \enquote{\textgr{άκρη}}).

    Then
    \begin{equation*}
      \synin: \syn\kappa \synimplies \syn\alpha \synimplies \syn\omicron.
    \end{equation*}

    With the obvious interpretation of \( \synin \), and with the formal details deferred to \cref{ex:fol_to_hol_homomorphism/hypergraphs}, a \hyperref[def:hol_homomorphism]{higher-order structure homomorphism} \( f: G \to H \) in this theory must satisfy
    \begin{equation*}
      v \in \mscrE_G(e) \T{if and only if} f(v) \in \mscrE_H(f(e)).
    \end{equation*}

    This is strictly stronger than \cref{def:hypergraph/morphism}. Thus, without alteration, we cannot use the semantics of higher-order logic to do deduce the hypergraph homomorphisms we have defined in \cref{def:hol_homomorphism}.

    \thmitem{rem:logical_theory_of_graphs/directed_graphs} \hyperref[def:directed_graph]{Directed graphs} allow a characterization using only one-sorted \hyperref[def:first_order_logic]{first-order logic}, as the empty \hyperref[def:fol_theory]{theory} over the signature with one infix predicate symbol, \( {\latimplies} \), determining whether there is an arc between the given endpoints.

    A \hyperref[def:fol_homomorphism]{first-order homomorphism} \( f: G \to H \) must then preserve this predicate, i.e.
    \begin{equation*}
      u \to v \T{implies} f(u) \to f(v).
    \end{equation*}

    This definition coincides with the directed graph homomorphisms from \cref{def:directed_graph/morphism}.

    \thmitem{rem:logical_theory_of_graphs/undirected_graphs} We can extend this theory with the symmetry axiom \eqref{eq:def:binary_relation/symmetric} to encompass \hyperref[def:undirected_graph]{undirected graphs}.
  \end{thmenum}
\end{remark}

\begin{remark}\label{rem:simple_graphs}
  Whether simple graphs are allowed have loops again depends on the authors.

  Defining \enquote{directed graphs} as pairs \( (V, A) \), where \( A \subseteq V \times V \), without further restrictions, is common. It is done in
  \cite[10]{Harary1969GraphTheory},
  \cite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory},
  \cite[21]{GondranMinoux1984GraphsAndAlgorithms},
  \cite[571]{Stanley2012EnumerativeCombinatoricsVol1},
  \cite[def. 10.1.2]{Rosen2019DiscreteMathematics},
  \cite[190]{Erickson2019Algorithms},
  \cite[279]{ЕмеличевИПр1990ТеорияГрафов},
  \cite[277]{БелоусовТкачёв2004ДискретнаяМатематика} and
  \cite[6]{Мирчев2001Графи}.

  \Textcite[\S 7.1.5]{Новиков2013ДискретнаяМатематика} also provides the same definition, however he forbids loops unless explicitly mentioned. \Textcite[28]{Diestel2017GraphTheory} uses the term \enquote{oriented graph} for what we call \enquote{simple directed graph}.

  For undirected graphs we instead have several conventions:
  \begin{itemize}
    \item Some of the aforementioned authors, namely \textcite[21]{GondranMinoux1984GraphsAndAlgorithms} and \textcite[277]{БелоусовТкачёв2004ДискретнаяМатематика}, define directed graphs as \enquote{symmetric} undirected graphs, in which \( (u, v) \) is an arc whenever \( (v, u) \) is. This definition allows loops since the directed counterparts do.

    \item Other aforementioned authors, namely \textcite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory}, \textcite[571]{Stanley2012EnumerativeCombinatoricsVol1}, \textcite[def. 10.1.1]{Rosen2019DiscreteMathematics}, \textcite[190]{Erickson2019Algorithms}, and \textcite[7]{Мирчев2001Графи} define undirected graphs as pairs \( (V, E) \), where \( E \) consists of one-element or two-element subsets of \( V \). This definition again allows loops.

    \item \Textcite[9]{Harary1969GraphTheory}, \textcite[2]{Diestel2017GraphTheory}, \textcite[1]{Bollobás1998ModernGraphTheory}, \textcite[9]{ЕмеличевИПр1990ТеорияГрафов} and \textcite[\S 7.1.2]{Новиков2013ДискретнаяМатематика} define directed graphs as pairs \( (V, E) \), where \( E \) consists of two-element subsets of \( V \). This explicitly forbid loops.

    \item \Textcite[377]{Knuth1997ArtVol1} explicitly forbids loops in graphs, however he defines \enquote{graphs} as \enquote{a set of points together with a set of lines}.
  \end{itemize}

  \Textcite[def. 1.1.2]{Knauer2019AlgebraicGraphTheory}, \textcite[571]{Stanley2012EnumerativeCombinatoricsVol1} and \textcite[def. 10.1.1]{Rosen2019DiscreteMathematics} use the adjective \enquote{simple} to refer to (multi)graphs without multiple edges, without referring to loops. So do \textcite[2]{Diestel2017GraphTheory} and \textcite[22]{Bollobás1998ModernGraphTheory}, however their definitions of graphs forbid loops by default. \Textcite[22]{GondranMinoux1984GraphsAndAlgorithms}, \textcite[191]{Erickson2019Algorithms}, \textcite[17]{ЕмеличевИПр1990ТеорияГрафов} and \textcite[12]{Мирчев2001Графи} use \enquote{simple} to additionally forbid loops.

  We prefer our terminology to be as unambiguous as possible, for which reason we use \enquote{simple} in the latter sense and, if loops are allowed, we mention it explicitly.
\end{remark}

\begin{proposition}\label{thm:graph_isomorphisms}
  We characterize \hyperref[def:morphism_invertibility/isomorphism]{categorical isomorphisms} in the respective categories of graphs.

  \begin{thmenum}
    \thmitem{thm:graph_isomorphisms/directed_multigraph} Given \hyperref[def:directed_multigraph]{directed multigraphs} \( G \) and \( H \), the \hyperref[def:directed_multigraph/morphism]{directed multigraph homomorphism}
    \begin{align*}
      &\varphi_V: V_G \to V_H, \\
      &\varphi_A: A_G \to A_H,
    \end{align*}
    is a categorical isomorphism if and only if both \( \varphi_V \) and \( \varphi_A \) are bijective.

    Furthermore, the componentwise inverse \( (\varphi_V^{-1}, \varphi_A^{-1}) \) of \( (\varphi_V, \varphi_A) \) is also a homomorphism, and it is the two-sided inverse of \( (\varphi_V, \varphi_A) \) in the category of directed multigraphs.

    \thmitem{thm:graph_isomorphisms/simple_directed} Given \hyperref[def:directed_graph]{simple directed graphs} \( G \) and \( H \), with loops allowed, the \hyperref[def:directed_graph/morphism]{directed graph homomorphism} \( \varphi: V_G \to V_H \) is a categorical isomorphism if and only if it is bijective and satisfies
    \begin{equation}\label{eq:thm:graph_isomorphisms/simple_directed}
      (u, v) \in A_G \T{if and only if} (\varphi(u), \varphi(v)) \in A_H.
    \end{equation}

    Furthermore, the inverse function \( \varphi^{-1} \) of \( \varphi \) is also a homomorphism, it satisfies \eqref{eq:thm:graph_isomorphisms/simple_directed}, and it is the two-sided inverse of \( \varphi \) in the category of simple directed graphs.

    \thmitem{thm:graph_isomorphisms/hypergraphs} Given \hyperref[def:hypergraph]{hypergraphs} \( G \) and \( H \), the \hyperref[def:hypergraph/morphism]{hypergraph homomorphism}
    \begin{align*}
      &\varphi_V: V_G \to V_H, \\
      &\varphi_E: E_G \to E_H,
    \end{align*}
    is a categorical isomorphism if and only if both \( \varphi_V \) and \( \varphi_E \) are bijective.

    Furthermore, the componentwise inverse \( (\varphi_V^{-1}, \varphi_E^{-1}) \) of \( (\varphi_V, \varphi_E) \) is also a homomorphism, and it is the two-sided inverse of \( (\varphi_V, \varphi_E) \) in the category of directed multigraphs.

    \thmitem{thm:graph_isomorphisms/simple_undirected} Given \hyperref[def:undirected_graph]{simple undirected graphs} \( G \) and \( H \), with loops allowed, the \hyperref[def:undirected_graph/morphism]{undirected graph homomorphism} \( \varphi: V_G \to V_H \) is a categorical isomorphism if and only if it is bijective and satisfies
    \begin{equation}\label{eq:thm:graph_isomorphisms/simple_undirected}
      \set{ u, v } \in E_G \T{if and only if} \set{ \varphi(u), \varphi(v) } \in E_H.
    \end{equation}
  \end{thmenum}

  Furthermore, the inverse function \( \varphi^{-1} \) of \( \varphi \) is also a homomorphism, it satisfies \eqref{eq:thm:graph_isomorphisms/simple_undirected}, and it is the two-sided inverse of \( \varphi \) in the category of simple undirected graphs.
\end{proposition}
\begin{comments}
  \item The category of hypergraphs is discussed in detail by \textcite{DörflerWaller1980Hypergraphs}.

  \textcite[def. 1.4.1]{Knauer2019AlgebraicGraphTheory} give \eqref{eq:thm:graph_isomorphisms/simple_directed} as a definition of graph isomorphism.

  \textcite[3]{Diestel2017GraphTheory}, \textcite[3]{Bollobás1998ModernGraphTheory}, \textcite[def. 1.4.3]{Knauer2019AlgebraicGraphTheory}, \textcite[def. 10.3.1]{Rosen2019DiscreteMathematics}, \textcite[13]{ЕмеличевИПр1990ТеорияГрафов}, \textcite[def. 5.14]{БелоусовТкачёв2004ДискретнаяМатематика} and \textcite[\S 7.1.6]{Новиков2013ДискретнаяМатематика} define simple undirected graph isomorphisms via \eqref{eq:thm:graph_isomorphisms/simple_undirected}, however avoid discussing isomorphisms for other kinds of graphs.
\end{comments}
\begin{proof}
  \SubProofOf{thm:graph_isomorphisms/directed_multigraph}

  \SufficiencySubProof* Suppose that \( (\varphi_V, \varphi_A): G \to H \) is a categorical isomorphism. Clearly it is a homomorphism. We must show that both \( \varphi_V \) and \( \varphi_A \) are bijective.

  We have two forgetful functors into \( \cat{Set} \) --- one for vertices and functions between them, the other one for arcs and functions between them. Hence, by \cref{thm:def:functor_invertibility/preserves_inverses}, both \( \varphi_V \) and \( \varphi_A \) are set isomorphisms, and, by \cref{thm:function_invertibility_categorical/fully_invertible}, both are bijective.

  \NecessitySubProof* Suppose that \( (\varphi_V, \varphi_A): G \to H \) is a homomorphism and that both \( \varphi_V \) and \( \varphi_A \) are bijective. We will show that \( (\varphi_V^{-1}, \varphi_A^{-1}) \) is a homomorphism from \( H \) to \( G \).

  For any arc \( e \) in \( H \), we have
  \begin{equation*}
    \varphi_V(h_G(\varphi^{-1}_A(e)))
    \reloset {\eqref{eq:def:directed_multigraph/morphism/head}} =
    h_H(\varphi_A(\varphi^{-1}_A(e)))
    =
    h_H(e),
  \end{equation*}
  hence
  \begin{equation*}
    h_G(\varphi^{-1}_A(e))
    =
    \varphi^{-1}(\varphi(h_G(\varphi^{-1}_A(e))))
    =
    \varphi^{-1}(h_H(e)).
  \end{equation*}

  We can analogously show that
  \begin{equation*}
    t_G(\varphi^{-1}_A(e)) = \varphi^{-1}_V(t_H(e)).
  \end{equation*}

  Thus, the pair \( (\varphi_V^{-1}, \varphi_A^{-1}) \) satisfies both \eqref{eq:def:directed_multigraph/morphism/head} and \eqref{eq:def:directed_multigraph/morphism/tail}, which makes it a directed multigraph homomorphism.

  Composing \( (\varphi_V, \varphi_A) \) and \( (\varphi_V^{-1}, \varphi_A^{-1}) \) (componentwise), we obtain identity homomorphisms on either \( G \) or \( H \) depending on the order of composition. Therefore, \( (\varphi_V, \varphi_A) \) is fully invertible, that is, a categorical isomorphism.

  \SubProofOf{thm:graph_isomorphisms/simple_directed}

  \SufficiencySubProof* Suppose that \( \varphi_V: V_G \to V_H \) is a categorical isomorphism between \( G \) and \( H \). Again, by \cref{thm:def:functor_invertibility/preserves_inverses}, \( \varphi_V \) is an isomorphism between \( V_G \) and \( V_H \) in the category of sets, that is, a bijective function.

  Let \( \psi: V_G \to V_H \) be the categorical inverse of \( f \). Then it is the inverse of \( \varphi \) in \( \cat{Set} \), which due to the uniqueness of two-sided inverses implies that \( \psi \) is the inverse function \( \varphi^{-1} \) of \( \varphi \).

  Then, if \( (\varphi(u), \varphi(v)) \) is an arc of \( H \) for some vertices \( u \) and \( v \),
  \begin{equation*}
    \parens[\big]{ \varphi^{-1}(\varphi(u)), \varphi^{-1}(\varphi(v)) } = (u, v)
  \end{equation*}
  is an edge of \( G \). Therefore, \eqref{eq:thm:graph_isomorphisms/simple_directed} holds.

  \NecessitySubProof* Suppose that \( \varphi: V_G \to V_H \) is a bijective homomorphism from \( G \) to \( H \) and that \eqref{eq:thm:graph_isomorphisms/simple_directed} holds.

  For any arc \( (u, v) \) in \( H \), we have
  \begin{equation*}
    (u, v) = \parens[\big]{ \varphi(\varphi^{-1}(u)), \varphi(\varphi^{-1}(v)) }.
  \end{equation*}
  and thus, by \eqref{eq:thm:graph_isomorphisms/simple_directed}, \( (\varphi^{-1}(u), \varphi^{-1}(v)) \) is an arc in \( G \).

  Therefore, \( \varphi^{-1} \) is itself a homomorphism from \( H \) to \( G \), and it clearly satisfies \eqref{eq:thm:graph_isomorphisms/simple_directed}. When composed with \( \varphi \), we obtain the identity on either \( V_G \) or \( V_H \) depending on the order of composition, hence \( \varphi^{-1} \) is the two-sided inverse of \( \varphi \).

  \SubProofOf{thm:graph_isomorphisms/hypergraphs}

  \SufficiencySubProof* If \( (\varphi_V, \varphi_E): G \to H \) is a categorical isomorphism of hypergraphs, we can prove that both \( \varphi_V \) and \( \varphi_E \) are bijective analogously to the case for directed multigraphs.

  \NecessitySubProof* Suppose that \( (\varphi_V, \varphi_E): G \to H \) is a homomorphism and that both \( \varphi_V \) and \( \varphi_E \) are bijective.

  For any edge \( e \) of \( H \), we have
  \begin{equation*}
    \mscrE_H(e)
    =
    \mscrE_H\parens[\big]{ \varphi_E(\varphi_E^{-1}(e)) }
    \reloset {\eqref{eq:def:hypergraph/morphism}} =
    \set{ \varphi(u) \given u \in \mscrE_G(\varphi_E^{-1}(e)) }.
  \end{equation*}

  Then
  \begin{equation*}
    \set{ \varphi^{-1}(v) \given v \in \mscrE_H(e) }
    =
    \set{ \varphi^{-1}(\varphi(u)) \given u \in \mscrE_G(\varphi_E^{-1}(e)) }
    =
    \mscrE_G(\varphi_E^{-1}(e)).
  \end{equation*}

  Therefore, the pair \( (\varphi_V^{-1}, \varphi_E^{-1}) \) satisfies \eqref{eq:def:hypergraph/morphism}, which makes it a hypergraph homomorphism.

  Analogously to the case of directed multigraphs, we conclude that \( (\varphi_V, \varphi_E) \) is a categorical isomorphism with inverse \( (\varphi_V^{-1}, \varphi_E^{-1}) \).

  \SubProofOf{thm:graph_isomorphisms/simple_undirected} Both sufficiency and necessity can be proven similarly to the case of simple directed graphs.
\end{proof}

\begin{proposition}\label{thm:subgraphs}
  We show that the \hyperref[def:categorical_subobject]{categorical subobjects} in the respective categories of graphs correspond exactly to the subgraphs we have defined. More specifically, the following statement holds for \hyperref[def:directed_multigraph]{directed multigraphs}, \hyperref[def:directed_graph]{simple directed graphs}, \hyperref[def:hypergraph]{hypergraphs} and \hyperref[def:undirected_graph]{simple undirected graphs}:
  \begin{displayquote}
    More specifically, if \( H \) and \( K \) are subgraphs \( G \), the inclusions \( \iota_H: H \to G \) and \( \iota_K: K \to G \) induce the same categorical subobject of \( G \) if and only if \( H = K \).
  \end{displayquote}
\end{proposition}
\begin{proof}
  We will give a proof for directed graphs. The other cases are similar.

  Suppose that \( H \) and \( K \) are subgraphs of \( G \) and let \( \iota_H: H \to G \) and \( \iota_K: K \to G \) be their inclusions. Both \( \iota_H \) and \( \iota_K \) induce categorical subobjects because monomorphisms --- \( \varphi \bincirc \iota_H = \psi \bincirc \iota_K \) is equivalent to
  \begin{equation*}
    \varphi(v) = \varphi(\iota_H(v)) = \psi(\iota_K(v)) = \psi(v).
  \end{equation*}

  \SufficiencySubProof* Suppose that \( \iota_H \) and \( \iota_K \) induce the same subobjects. Then by definition there exists some isomorphism \( \varphi: H \to K \) such that \( \iota_H = \iota_K \bincirc \varphi \). Thus, for every vertex \( v \) in \( H \),
  \begin{equation*}
    v = \iota_H(v) = \iota_K(\varphi(v)) = \varphi(v),
  \end{equation*}
  so \( \varphi \) is the inclusion of \( V_H \) into \( V_K \). Since \( \varphi \) is bijective, we conclude that \( V_H = V_K \).

  For the arcs, we have
  \begin{equation*}
    \underbrace{(\iota_H(u), \iota_H(v)) \in A_G}_{(u, v) \in A_H} \T{if and only if} \underbrace{(\iota_K(\varphi(u)), \iota_K(\varphi(v))) \in A_G}_{(\varphi(u), \varphi(v)) \in A_K},
  \end{equation*}
  therefore \( A_H = A_K \).

  We conclude that \( H \) and \( K \) coincide.

  \NecessitySubProof* Conversely, if \( H = K \), then \( \iota_H = \iota_K \), so the induced subobjects obviously coincide.
\end{proof}

\begin{definition}\label{def:multigraph_orientation}\mcite[def. 6.2.1; def. 1.1.2]{Knauer2019AlgebraicGraphTheory}
  We call the \hyperref[def:directed_multigraph]{directed multigraph} \( D = (V, A, h, t) \) an \term[ru=ориентация (\cite[32]{ЕмеличевИПр1990ТеорияГрафов})]{orientation} of the \hyperref[def:hypergraph/multigraph]{undirected multigraph} \( G = (U, E, \mscrE) \) if \( U = V \), \( A = E \), and, for every arc \( e \in A \), we have \( \mscrE(e) = \set{ h(e), t(e) } \).

  We call \( G \) the \term{underlying undirected (multi)graph} of \( D \).
\end{definition}
\begin{comments}
  \item This definition also applies to simple graphs via the inclusion functors \hyperref[def:graph_functors/directed_inclusion]{\( I_D \)} and \hyperref[def:graph_functors/undirected_inclusion]{\( I_U \)}.
\end{comments}

\begin{definition}\label{def:graph_functors}\mimprovised
  Different kinds of graphs are related via the following functors (shown graphically in \cref{fig:def:graph_functors}):
  \begin{thmenum}
    \thmitem{def:graph_functors/directed_forgetful} The directed \hyperref[def:concrete_category]{forgetful functor}:
    \begin{flalign*}
      &U_D: \hyperref[def:directed_multigraph/category]{\T*{Directed multigraphs}} \to \hyperref[def:directed_graph/category]{\T*{Simple directed graphs with loops}}, &&\\
      &U_D(V, A, h, t) \coloneqq \parens[\big]{ V, \set[\big]{ \parens[\big]{ h(a), t(a) } \given* a \in A } }, &&\\
      &U_D(\varphi_V, \varphi_A) \coloneqq \varphi_V.
    \end{flalign*}

    \thmitem{def:graph_functors/directed_inclusion} A \hyperref[def:category_adjunction]{left adjoint}:
    \begin{flalign*}
      &I_D: \hyperref[def:directed_graph/category]{\T*{Simple directed graphs}} \to \hyperref[def:directed_multigraph/category]{\T*{Directed multigraphs}}, &&\\
      &I_D(V, A) \coloneqq \parens[\big]{ V, A, (u, v) \mapsto u, (u, v) \mapsto v }, &&\\
      &I_D(\varphi) \coloneqq \parens[\big]{ \varphi, (u, v) \mapsto \parens[\big]{ \varphi(u), \varphi(v) } }.
    \end{flalign*}

    \thmitem{def:graph_functors/undirected_forgetful} A similar forgetful functor for undirected graphs:
    \begin{flalign*}
      &U_U: \hyperref[def:hypergraph/category]{\T*{Undirected multigraphs}} \to \hyperref[def:undirected_graph/category]{\T*{Simple undirected graphs with loops}}, &&\\
      &U_U(V, E, \mscrE) \coloneqq \parens[\big]{ V, \set[\big]{ \mscrE(e) \given* e \in E } }, &&\\
      &U_U(\varphi, \varphi_E) \coloneqq \varphi.
    \end{flalign*}

    \thmitem{def:graph_functors/undirected_inclusion} A left adjoint:
    \begin{flalign*}
      &I_U: \hyperref[def:undirected_graph/category]{\T*{Simple undirected graphs}} \to \hyperref[def:hypergraph/category]{\T*{Undirected multigraphs}}, &&\\
      &I_U(V, E) \coloneqq \parens[\big]{ V, E, e \mapsto e }, &&\\
      &I_U(\varphi) \coloneqq \parens[\big]{ \varphi, \set{ u, v } \mapsto \set{ \varphi(u), \varphi(v) } }.
    \end{flalign*}

    \thmitem{def:graph_functors/multi_forgetful} A functor that forgets the \hyperref[def:multigraph_orientation]{orientation} of a multigraph:
    \begin{flalign*}
      &U_M: \hyperref[def:directed_multigraph/category]{\T*{Directed multigraphs}} \to \hyperref[def:hypergraph/category]{\T*{Undirected multigraphs}}, &&\\
      &U_M(V, A, h, t) \coloneqq \parens[\big]{ V, A, e \mapsto \set{ h(e), t(e) } }, &&\\
      &U_M(\varphi_V, \varphi_A) \coloneqq (\varphi_V, \varphi_A).
    \end{flalign*}

    \thmitem{def:graph_functors/simple_forgetful} A functor that forgets the orientation of a simple graph:
    \begin{flalign*}
      &U_S: \hyperref[def:directed_graph/category]{\T*{Simple directed graphs}} \to \hyperref[def:undirected_graph/category]{\T*{Simple undirected multigraphs}}, &&\\
      &U_S(V, A) \coloneqq \parens[\big]{ V, A, (u, v) \mapsto \set{ u, v } }, &&\\
      &U_S(\varphi) \coloneqq \varphi.
    \end{flalign*}

    Note that opposite arcs map to the same edge.

    \thmitem{def:graph_functors/simple_doubling} A right adjoint to \( U_S \), which doubles each edge to produce arcs in both directions:
    \begin{flalign*}
      &D_S: \hyperref[def:undirected_graph/category]{\T*{Simple undirected graphs}} \to \hyperref[def:directed_multigraph/category]{\T*{Simple directed graphs}}, &&\\
      &D_S(V, E) \coloneqq \parens[\big]{ V, \set[\big]{ (u, v) \in V^2 \given \set{ u, v } \in E } }, &&\\
      &D_S(\varphi) \coloneqq \varphi.
    \end{flalign*}
  \end{thmenum}
\end{definition}
\begin{comments}
  \item \Cref{ex:def:category_adjunction/us_ds} discusses the doubling functor from \cref{def:graph_functors/simple_doubling}.
  \item We can introduce a doubling functor for multigraphs similar to \hyperref[def:graph_functors/simple_doubling]{\( D_S \)} that introduces two opposite copies of each edge in an undirected multigraph.

  Unfortunately, that would require choosing one vertex as the head and the other as a tail, and thus the functor depends on a \hyperref[def:choice_function]{choice function}. This choice function is unnecessary for simple graphs because ordered pairs having a first and second element, while multigraphs have abstract objects as arcs.

  Furthermore, even given a canonical choice function, this functor would not be a right inverse to \hyperref[def:graph_functors/multi_forgetful]{\( U_M \)} --- instead, we would obtain an undirected multigraph with twice as many edges as the original.

  We avoid introducing such a functor altogether.
\end{comments}

\begin{remark}\label{rem:graph_of_unspecified_type}
  We will henceforth use the phrase \enquote{graph of unspecified type} to refer to any of the four kinds of (un)directed (multi)graphs. We will mostly avoid working with hypergraphs, hence we will be explicit about them and say, for example, \enquote{graph of unspecified type of hypergraph}.

  The inclusion functors \hyperref[def:graph_functors/directed_inclusion]{\( I_D \)} and \hyperref[def:graph_functors/undirected_inclusion]{\( I_U \)} allow all definitions for multigraphs to apply to simple graphs. The transition between directed and undirected graphs is more complicated, but we will nonetheless utilize \hyperref[def:graph_functors/multi_forgetful]{\( U_M \)}, \hyperref[def:graph_functors/simple_forgetful]{\( U_S \)} and \hyperref[def:graph_functors/simple_doubling]{\( D_S \)} when necessary.

  These functors will be used implicitly, but when increased attention is required, we use these functors explicitly. See \cref{thm:graph_coloring_as_homomorphism} for an example.
\end{remark}

\begin{definition}\label{def:induced_subgraph}\mcite[def. 1.4.8]{Knauer2019AlgebraicGraphTheory}
  In an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type} \( G \), for every set \( U \) of vertices, we define the \term[ru=порождённый (подграф) (\cite[17]{ЕмеличевИПр1990ТеорияГрафов}), bg=породен (подграф) (\cite[18]{Мирчев2001Графи})]{induced subgraph} \( G[U] \) as the unique subgraph with vertex set \( U \) that contains all arcs/edges of \( G \) with endpoints in \( U \).
\end{definition}

\begin{definition}\label{def:graph_incidence}\mcite[def. 1.1.1]{Knauer2019AlgebraicGraphTheory}
  In an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type}, we say that the vertex \( v \) and the arc/edge \( e \) are \term[bg=инцидентни (ребра) (\cite[7]{Мирчев2001Графи}), ru=инцидентные (рёбра) (\cite[9]{ЕмеличевИПр1990ТеорияГрафов})]{incident} of \( v \) is an endpoint of \( e \).
\end{definition}

\begin{definition}\label{def:graph_adjacency}\mcite[def. 10.2.1; def. 10.2.2]{Rosen2019DiscreteMathematics}
  In an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type}, if two vertices (resp. arcs/edges) are \hyperref[def:graph_incidence]{incident} to a common arc/edge (resp. vertex), we say that they are \term[bg=съседни (върхове/дъги/ребра) (\cite[7]{Мирчев2001Графи}), ru=смежные (вершины/рёбра) (\cite[9]{ЕмеличевИПр1990ТеорияГрафов})]{adjacent}.
\end{definition}

\begin{definition}\label{def:graph_independent_set}\mcite[777]{Rosen2019DiscreteMathematics}
  We say that a subset of the vertices (resp. arcs/edges) of an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type} is an \term[bg=независимо множество (\cite[103]{Мирчев2001Графи})]{independent set} if no two vertices (resp. arcs/edges) in it are \hyperref[def:graph_adjacency]{adjacent}.
\end{definition}

\begin{remark}\label{rem:trivial_graph}
  Unlike the \hyperref[def:group/trivial]{trivial group} \( \set{ e } \) or \hyperref[def:module/trivial]{trivial \( R \)-module} \( \set{ 0 } \), which are unique up to an isomorphism, there is no trivial graph in the sense of \cref{def:trivial_object}.

  Every graph has a subgraph with \hyperref[def:graph_cardinality/order]{order} zero, and hence up to an isomorphism we have an order-zero graph (for every kind of graph discussed here).

  Another unambiguous concept is that of an edgeless graph. Every \hyperref[def:complete_graph]{complete graph} has \( 2^n \) edgeless subgraphs (one for each set of edges).
\end{remark}

\paragraph{Cardinalities in graphs}

\begin{definition}\label{def:graph_cardinality}
  Graphs have the following notions of \hyperref[thm:cardinality_existence]{cardinality}:
  \begin{thmenum}
    \thmitem{def:graph_cardinality/order}\mcite[2]{Diestel2017GraphTheory} We define the \term[ru=порядок (\cite[9]{ЕмеличевИПр1990ТеорияГрафов})]{order} \( \ord(G) \) of an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type} \( G \) as the (cardinal) number of vertices.

    For a simple graph, finitely many vertices imply finitely many arcs/edges, which justifies terminology like \enquote{finite simple graph}. For multigraphs, however, we will prefer being more concrete and use \enquote{finite-order (multi)graph}.

    \thmitem{def:graph_cardinality/directed_degree}\mimprovised We define the \term[bg=полустепен на изхода (\cite[8]{Мирчев2001Графи}), ru=полустепень исхода (\cite[283]{ЕмеличевИПр1990ТеорияГрафов}), en=out-degree (\cite[def. 1.1.7]{Knauer2019AlgebraicGraphTheory})]{out-degree} \( \deg_{\op{out}}(v) \) (resp. \term[bg=полустепен на входа (\cite[8]{Мирчев2001Графи}), ru=полустепень захода (\cite[283]{ЕмеличевИПр1990ТеорияГрафов}), en=in-degree (\cite[def. 1.1.8]{Knauer2019AlgebraicGraphTheory})]{in-degree} \( \deg_{\op{in}}(v) \)) of a vertex \( v \) in a \hyperref[def:directed_multigraph]{directed (multi)graph} as the (cardinal) number of arcs starting (resp. ending) at \( v \).

    We define the \term[bg=степен (\cite[8]{Мирчев2001Графи}), ru=степень (\cite[283]{ЕмеличевИПр1990ТеорияГрафов})]{total degree} \( \deg(v) \) of \( v \) as the sum of the two.

    \thmitem{def:graph_cardinality/undirected_degree}\mimprovised In an undirected graph or hypergraph, consider the \hyperref[def:multiset]{multiset} of all edges incident to \( v \), with loops having multiplicity \( 2 \) and all other edges having multiplicity \( 1 \).

    We define the \term{degree} \( \deg(v) \) as the \hyperref[def:multiset_cardinality]{cardinality} of this multiset.

    \thmitem{def:graph_cardinality/local}\mcite[210]{Diestel2017GraphTheory} We say that a graph is \term{locally finite} if the degree of any vertex is finite.
  \end{thmenum}
\end{definition}
\begin{comments}
  \item We similarly use other adjectives based on cardinality, e.g. (locally) denumerable graph; see \cref{rem:cardinality_auxiliary_terminology}.
\end{comments}

\begin{proposition}\label{thm:degree_of_undirected_counterpart}
  The degree, in the sense of \cref{def:graph_cardinality/directed_degree}, of a vertex \( v \) in a \hyperref[def:directed_multigraph]{directed multigraph} \( G \), is equal to the degree, in the sense of \cref{def:graph_cardinality/undirected_degree}, of \( v \) in the undirected counterpart \( \hyperref[def:graph_functors/multi_forgetful]{U_M}(G) \) of \( G \).
\end{proposition}
\begin{comments}
  \item When proving statements about vertex degrees, it thus makes sense to prove it for undirected multigraphs, since it will then also automatically apply to directed graphs.
\end{comments}
\begin{proof}
  We have adjusted our definitions so that this holds.
\end{proof}

\begin{remark}\label{rem:counting_loops_twice}
  Loops may contribute either \( 1 \) or \( 2 \) towards the degree of a graph. This choice simplifies \cref{thm:sum_of_endpoint_degrees} and hence \cref{thm:sum_of_graph_degrees}, as well as \cref{def:graph_cycle}, among others.

  We describe here several conventions.
  \begin{itemize}
    \item \Textcite[27]{ЕмеличевИПр1990ТеорияГрафов}, who diligently distinguish between directed and undirected graphs, defines degrees in (simple) directed graphs as sums of out-degrees and in-degrees, like we do, and in undirected graphs they, like us, count loops twice.

    In chapter XI, dedicated to hypergraphs, they define the degree of a vertex as the number of incident hyperedges without mentioning how loops are handled. There is no reason to believe that they would generalize from undirected graphs without counting loops twice.

    \item \Textcite[277]{БелоусовТкачёв2004ДискретнаяМатематика}, who also give separate definitions for directed and undirected graphs, also define degrees in (simple) directed graphs as sums of out-degrees and in-degrees, but for undirected graphs they avoid counting loops twice.

    \item \Textcite[def. 10.2.3]{Rosen2019DiscreteMathematics}, who also gives separate definitions between directed and undirected graphs, defines degrees for (simple) undirected graphs by explicitly counting loops twice, but for directed graphs they only define the in-degrees and out-degrees without the \enquote{total} degrees.

    \item \Textcite[191]{Erickson2019Algorithms} and \textcite[8]{Мирчев2001Графи} define degrees without counting loops twice for both directed and undirected graphs.

    \item \Textcite[def. 1.1.7]{Knauer2019AlgebraicGraphTheory} avoids defining \enquote{total} degrees for directed graphs, leaving only out-degrees and in-degrees. For undirected graphs, however, in \cite[def. 1.1.8]{Knauer2019AlgebraicGraphTheory} they define the degree of a vertex \( v \) as the number of edges incident to \( v \), which without our multiset trick only counts loops once.

    \item \Textcite[372]{Knuth1997ArtVol1} also defines out-degrees and in-degrees, but avoids defining degrees for both directed and undirected graphs.

    \item \Textcite[8]{Bollobás1998ModernGraphTheory} hints that loops should be counted twice.
  \end{itemize}
\end{remark}

\begin{definition}\label{def:isolated_vertex}\mcite[5]{Diestel2017GraphTheory}
  We say that a vertex in an \hyperref[rem:graph_of_unspecified_type]{(hyper)graph of unspecified type} is \term[bg=изолирани (върхове) (\cite[8]{Мирчев2001Графи}), ru=изолированные (вершины) (\cite[26]{ЕмеличевИПр1990ТеорияГрафов})]{isolated} if its \hyperref[def:graph_cardinality/directed_degree]{total degree} is zero.
\end{definition}

\begin{lemma}\label{thm:sum_of_endpoint_degrees}
  Fix an undirected multigraph \( G = (V, E, \mscrE) \). Fix an edge \( e \) and consider the graph \( G = (V, E \setminus \set{ e }, \mscrE) \). Denote by \( \deg'(v) \) the vertex degree in \( G' \).

  Then
  \begin{equation}\label{eq:thm:sum_of_endpoint_degrees}
    \sum_{v \in \mscrE(e)} \deg(v) = \sum_{v \in \mscrE(e)} \deg'(v) + 2.
  \end{equation}
\end{lemma}
\begin{proof}
  If \( e \) is a loop at \( v \), by definition we have
  \begin{equation*}
    \sum_{w \in \mscrE(e)} \deg(w) = \deg(v) = \deg'(v) + 2 = \sum_{w \in \mscrE(e)} \deg'(w) + 2.
  \end{equation*}

  If \( e \) is an edge between \( u \) and \( v \), then
  \begin{equation*}
    \sum_{w \in \mscrE(e)} \deg(w)
    =
    \deg(u) + \deg(v)
    =
    \deg'(u) + 1 + \deg'(v) + 1
    =
    \sum_{w \in \mscrE(e)} \deg'(w) + 2.
  \end{equation*}
\end{proof}

\begin{proposition}\label{thm:sum_of_graph_degrees}
  In an \hyperref[rem:graph_of_unspecified_type]{graph of unspecified type} with finitely many arcs/edges, the sum of the degrees of all vertices is twice the number of arcs/edges.
\end{proposition}
\begin{comments}
  \item As long as a graph only has finitely many arcs/edges, only finitely many vertices have positive degree, so summing all degrees is justified.
  \item As a consequence, the sum of degrees if even. \Textcite[4]{Bollobás1998ModernGraphTheory} and \textcite[prop. 5.1]{ЕмеличевИПр1990ТеорияГрафов} call this consequence the \enquote{handshaking lemma} because of the observation that the total number of hands shaken at a party is even. \Textcite[thm 10.2.1]{Rosen2019DiscreteMathematics} uses \enquote{handshaking lemma} for the proposition itself.
\end{comments}
\begin{proof}
  As per \cref{thm:degree_of_undirected_counterpart}, it is sufficient to only prove the statement for undirected multigraphs.

  We will use induction on the number \( n \) of arcs.

  \begin{itemize}
    \item If there are zero edges, then the degree of every vertex is zero, hence the sum of all degrees is also zero.

    \item Suppose that the statement holds for multigraphs with \( n \) arcs and consider an undirected multigraph \( G = (V, E, \mscrE) \) with \( n + 1 \) arcs.

    Fix any arc \( e \) and consider \( G' \coloneqq (V, E \setminus \set{ e }, \mscrE) \). Denote by \( \deg'(v) \) the degree of \( v \) in \( G' \). Then \cref{thm:sum_of_endpoint_degrees} implies that
    \begin{equation*}
      \overbrace{\sum_{v \in V} \deg(v) = \sum_{v \in V \setminus \mscrE(e)} \underbrace{\deg(v)}_{\deg'(v)} + \underbrace{\sum_{v \in \mscrE(e)} \deg(v)}_{\sum_{v \in \mscrE(e)} \deg'(v) + 2} - 2}^{2n \T*{by the inductive hypothesis}} + 2
      =
      2(n + 1)
    \end{equation*}
  \end{itemize}
\end{proof}

\begin{corollary}\label{thm:odd_degree_vertices}\mcite[prop. 1.2.1]{Diestel2017GraphTheory}
  In an \hyperref[rem:graph_of_unspecified_type]{graph of unspecified type} with finitely many arcs/edges, the number of vertices of odd \hyperref[def:graph_cardinality/directed_degree]{total degree} is even.
\end{corollary}
\begin{comments}
  \item This corollary, while by itself inappreciable, is essential for proving \fullref{thm:eulers_theorem_for_graphs}. It first appeared in \cite{Euler1741Bridges} by Leonhard Euler, considered the first paper on graph theory and discussed in \cref{ex:konigsberg_bridges}.
\end{comments}
\begin{proof}
  As per \cref{thm:degree_of_undirected_counterpart}, it is sufficient to consider some undirected multigraph \( G = (V, E, \mscrE) \).

  We have
  \begin{equation}\label{eq:thm:odd_degree_vertices/proof}
    \underbrace{\sum_{v \in V} \deg(v)}_{\T*{even by} \cref{thm:sum_of_graph_degrees}} = \underbrace{\sum_{\deg(v) \T*{is even}} \deg(v)}_{\T{sum of even numbers}} + \underbrace{\sum_{\deg(v) \T*{is odd}} \deg(v)}_{\T{sum of odd numbers}}.
  \end{equation}

  In order for \eqref{eq:thm:odd_degree_vertices/proof} to hold, there must be an even number of vertices of odd degree.
\end{proof}
