\section{Semimodules}\label{sec:semimodules}

\paragraph{Semimodules over semirings}

\begin{definition}\label{def:endomorphism_semiring}\mcite[example 1.14]{Golan1999Semirings}
  Let \( M \) be an \hyperref[def:additive_semigroup]{additive} \hyperref[def:binary_operation/commutative]{commutative} \hyperref[def:monoid]{monoid}. Then the \hyperref[def:endomorphism_monoid]{endomorphism monoid} \( \End(M) \) is a \hyperref[def:semiring]{semiring} with addition inherited from \( M \) and multiplication given by composition. The zero in \( \End(M) \) is the constant map to \( 0_M \) and the unit in \( \End(M) \) is the \hyperref[def:set_valued_map/identity]{identity map}.

  We call \( \End(M) \) as the \term{endomorphism semiring}.
\end{definition}
\begin{comments}
  \item It is important that \( M \) is commutative because otherwise \( \End(M) \) is not closed under addition.

  \item When \( M \) is an \hyperref[def:abelian_group]{abelian group}, then \( \End(M) \) is a \hyperref[def:ring]{ring} --- see \cref{def:endomorphism_ring}.

  \item We use the definition from \cite[example 1.14]{Golan1999Semirings}, who however introduces no special name for \( \End(M) \). We base the term \enquote{endomorphism semiring} on the better established concept of endomorphism rings, which we discuss in \cref{def:endomorphism_ring}.
\end{comments}
\begin{defproof}
  We must show that \( R \coloneqq \End(M) \) is indeed a semiring.

  \SubProof{Proof of closure under \( {+_R} \)} Suppose that \( \varphi \) and \( \psi \) are endomorphisms of \( M \). We need to show that \( \theta \coloneqq \varphi + \psi \) is also an endomorphism.

  \SubProof*{Proof that \( \theta(x + y) = \theta(x) + \theta(y) \)}
  \begin{align*}
    \theta(x + y)
    &=
    [\varphi + \psi](x + y)
    = \\ &=
    \varphi(x + y) + \psi(x + y)
    = \\ &=
    \varphi(x) + \varphi(y) + \psi(x) + \psi(y)
    = \\ &=
    \varphi(x) + \psi(x) + \varphi(y) + \psi(y)
    = \\ &=
    \theta(x) + \theta(y).
  \end{align*}

  \SubProof*{Proof that \( \theta(0_M) = 0_R \)}
  \begin{equation*}
    \theta(0_M)
    =
    [\varphi + \psi](0_M)
    =
    \varphi(0_M) + \psi(0_M)
    =
    0_R + 0_R
    =
    0_R.
  \end{equation*}

  \SubProofOf[def:binary_operation/associative]{associativity of \( {+_R} \)} Follows from associativity of \( {+_M} \).

  \SubProofOf[def:binary_operation/commutative]{commutativity of \( {+_R} \)} Follows from commutativity of \( {+_M} \).

  \SubProof{Proof that \( 0_R \) is neutral under \( {+_M} \)} We have
  \begin{equation*}
    (\varphi + 0_R)(x)
    =
    \varphi(x) + 0_R(x)
    =
    \varphi(x)
    =
    0_R(x) + \varphi(x)
    =
    (0_R + \varphi)(x).
  \end{equation*}

  \SubProofOf[def:binary_operation/associative]{associativity of \( \cdot_R \)} Follows from associativity of composition.

  \SubProof{Proof that \( \id \) is neutral under \( {\bincirc} \)}
  \begin{equation*}
    (\varphi \bincirc \id)(x)
    =
    \varphi(\id(x))
    =
    \varphi(x)
    =
    \id(\varphi(x))
    =
    (\id \bincirc \varphi)(x).
  \end{equation*}

  \SubProof{Proof that \( 0_R \) is absorbing}
  \begin{equation*}
    (\varphi \bincirc 0_R)(x)
    =
    \varphi(0_R(x))
    =
    \varphi(0_M)
    =
    0_M
    =
    0_R(x).
  \end{equation*}

  \SubProofOf[eq:def:semiring/theory/axioms/distributivity/left]{left distributivity}
  \begin{equation*}
    [\varphi \bincirc (\psi + \theta)](x)
    =
    \varphi(\psi(x) + \theta(x))
    =
    \varphi(\psi(x)) + \varphi(\theta(x))
    =
    [\varphi \bincirc \psi + \varphi \bincirc \theta](x).
  \end{equation*}

  \SubProofOf[eq:def:semiring/theory/axioms/distributivity/right]{right distributivity}
  \begin{equation*}
    [(\varphi + \psi) \bincirc \theta](x)
    =
    \varphi(\theta(x)) + \psi(\theta(x))
    =
    [\varphi \bincirc \theta + \psi \bincirc \theta](x).
  \end{equation*}
\end{defproof}

\begin{definition}\label{def:semimodule}\mcite[149]{Golan1999Semirings}
  An \( R \)-\term{semimodule} is an \term[en=action of ring on abelian group (\cite[156]{Aluffi2009Algebra})]{action} of the \hyperref[def:semiring]{semiring} \( R \) on a \hyperref[def:monoid/commutative]{commutative} \hyperref[def:additive_semigroup]{additive} \hyperref[def:monoid]{monoid} \( M \) --- in a sense to be defined below. Similarly to \hyperref[def:monoid_action]{monoid actions}, if \( R \) is not commutative, we distinguish between left and right \( R \)-semimodules.

  We will call the elements of \( R \) \term[en=scalars (in module) (\cite[192]{Bourbaki1998Algebra1to3})]{scalars}. Moreover, even though this usage is only established for \hyperref[def:vector_space]{vector spaces}, we will occasionally call the elements of \( M \) \term[ru=векторы (в модуле) (\cite[78]{Мальцев2009ОсновыЛинейнойАлгебры}), en=vectors (in free module) (\cite[308]{Aluffi2009Algebra})]{vectors}. See \cref{rem:vector/vector_space} for a general discussion of these terms.

  There are multiple ways to express the desired interaction between \( R \) and \( M \):
  \begin{thmenum}[series=def:semimodule]
    \thmitem{def:semimodule/action}\mimprovised A left \( R \)-semimodule is a \hyperref[def:semiring/morphism]{semiring homomorphism} from \( R \) to the \hyperref[def:endomorphism_semiring]{endomorphism semiring} \( \End(M) \). A right \( R \)-semimodule is a homomorphism from \( R \) to the \hyperref[def:semiring/opposite]{opposite semiring} \( \End(M)^\oppos \).

    This definition is concise and natural, but requires some unpacking in order to reflect common usage.

    \thmitem{def:semimodule/operation}\mcite[149]{Golan1999Semirings} A more explicit definition of a left \( R \)-semimodule involves a heterogeneous binary operation \( \cdot: R \times M \to M \) called \term{scalar multiplication} (\( \cdot: M \times R \to M \) for right \( R \)-semimodules) that satisfies (the \hyperref[def:fol_quantifier_closure]{universal closures} of) the following conditions:
    \begin{thmenum}
      \thmitem{def:semimodule/operation/scalar_multiplication_action} Scalar multiplication is a \hyperref[def:monoid_action]{monoid action} of the multiplicative monoid \( (R, \cdot_R) \) on \( M \). The following conditions correspond to \eqref{eq:def:monoid_action/family/identity} and \eqref{eq:def:monoid_action/family/compatibility}:
      \begin{align}
        1_R \cdot x           &= x, \label{eq:def:semimodule/operation/action/identity} \\
        (r \cdot_R s) \cdot x &= r \cdot (s \cdot x). \label{eq:def:semimodule/operation/action/compatibility}
      \end{align}

      The first condition \eqref{eq:def:semimodule/operation/action/identity} is redundant in \hyperref[def:module]{modules}.

      The second condition \eqref{eq:def:semimodule/operation/action/compatibility} can be regarded as a form of \hyperref[def:binary_operation/associative]{associativity}. For right \( R \)-semimodules, we must replace it with
      \begin{equation}\label{eq:def:monoid_action/family/compatibility_right}
        x \cdot (r \cdot_R s) = (x \cdot r) \cdot s.
      \end{equation}

      \thmitem{def:semimodule/operation/scalar_addition_distributivity} Scalar multiplication distributes over addition  of scalars:
      \begin{equation}\label{eq:def:semimodule/operation/scalar_addition_distributivity}
        (r +_R s) \cdot x = r \cdot x +_M s \cdot x.
      \end{equation}

      \thmitem{def:semimodule/operation/vector_addition_distributivity} Scalar multiplication distributes over addition of vector:
      \begin{equation}\label{eq:def:semimodule/operation/vector_addition_distributivity}
        r \cdot (x +_M y) = r \cdot x +_M r \cdot y.
      \end{equation}

      \thmitem{def:semimodule/operation/absorption} The zero scalar and zero vector are \hyperref[def:semiring_absorbing_element]{absorbing} with respect to scalar multiplication:
      \begin{align}
        0_R \cdot x &= 0_M, \label{eq:def:semimodule/operation/absorption/scalar} \\
        r \cdot 0_M &= 0_M. \label{eq:def:semimodule/operation/absorption/vector}
      \end{align}

      Both absorption axioms are redundant in \hyperref[def:module]{modules}.
    \end{thmenum}

    In practice, we use the same symbol for both scalar and vector addition, and we denote both multiplication operations via juxtaposition.
  \end{thmenum}

  Semimodules have the following metamathematical properties:
  \begin{thmenum}[resume=def:semimodule]
    \thmitem{def:semimodule/theory}\mimprovised We will present a (one-sorted) \hyperref[def:fol_equational_theory]{equational theory} for semimodules. An alternative is to use \hyperref[def:many_sorted_fol]{two-sorted logic}; see \cref{rem:two_sorted_semimodules} for a more thorough discussion.

    The heterogeneous scalar multiplication operation can be emulated via a family \( \seq{ \synscalar_r }_{r \in R} \) of unary \hyperref[def:fol_signature/function]{function symbols}, one for each scalar.

    This allows us to extend the \hyperref[def:additive_semigroup/theory]{theory of abelian groups} with the monoid action axioms
    \begin{align*}
      \qforall \synx (\synscalar_1(\synx) &\syneq \synx), \\
      \qforall \synx (\synscalar_{rs}(\synx) &\syneq \synscalar_r(\synscalar_s(\synx))), \\
    \intertext{the distributivity axioms}
      \qforall \synx (\synscalar_{r + s}(\synx) &\syneq \synscalar_r(\synx) \synplus \synscalar_s(\synx)), \\
      \qforall \synx \qforall \syny (\synscalar_r(\synx \synplus \syny) &\syneq \synscalar_r(\synx) \synplus \synscalar_r(\syny)), \\
    \intertext{and the absorption axioms}
      \qforall \synx (\synscalar_0(\synx) &\syneq \syn0), \\
      \qforall \synx (\synscalar_r(\syn0) &\syneq \syn0).
    \end{align*}

    These axioms are parameterized by \( r \) and \( s \), so in fact we may have infinitely many axioms.

    \thmitem{def:semimodule/morphism}\mcite[156]{Golan1999Semirings} We say that the function \( \varphi: M \to N \) between left \( R \)-semimodules is \( R \)-\term[bg=линейно преобразувание (между линейни пространства) (\cite[101]{Обрешков1962ВисшаАлгебра}), ru=линейный оператор (между векторными пространствами) (\cite[236]{Тыртышников2007ЛинейнаяАлгебра}), en=linear mapping (between modules) (\cite[def. 2.9.1]{Bourbaki1998Algebra1to3})]{linear} if it is \hyperref[def:additive_function]{additive} and \( R \)-\hyperref[def:homogeneous_function]{homogeneous}, i.e. it if preserves vector addition and scalar multiplication with all elements of \( R \). The semiring with respect to which \( \varphi \) is linear is important --- see \cref{ex:def:semimodule/complex_conjugate}.

    The \( R \)-linear functions between \( M \) and \( N \) are precisely the \hyperref[def:fol_homomorphism]{first-order homomorphisms}, which makes the concepts from \fullref{sec:first_order_structures} applicable.

    \thmitem{def:semimodule/category}\mimprovised For a fixed \hyperref[def:grothendieck_universe]{Grothendieck universe} \( \mscrU \) and a fixed semiring \( R \), the first-order theory of \( R \)-semimodules gives rise to a \hyperref[def:fol_theory_model_functor/obj]{category of \( \mscrU \)-small models}.

    \thmitem{def:semimodule/subobject}\mcite[150]{Golan1999Semirings} An \( R \)-\term[en=subsemimodule]{sub-semimodule} of \( M \) is a \hyperref[def:group/subobject]{subgroup} that is closed under scalar multiplication with all elements of \( R \). The semiring used for this closure is important --- see \cref{ex:def:semimodule/natural_number_span}.

    It follows from \cref{thm:fol_equational_theory_models/substructure} that the subgroups of \( M \) are precisely its \hyperref[def:fol_substructure/submodel]{first-order submodels}, and from \cref{thm:fol_categorical_subobjects} that they are its \hyperref[def:categorical_subobject]{categorical subobjects}.

    Furthermore, as a consequence of \cref{thm:fol_equational_theory_models/image}, the image of a linear map is a \( R \)-sub-semimodule of its codomain.

    \thmitem{def:semimodule/generated}\mcite[153]{Golan1999Semirings} We define the \( R \)-\term[ru=линейная оболочка (\cite[\S 3.2]{Тыртышников2007ЛинейнаяАлгебра}), en=linear span (\cite[35]{Knapp2016BasicAlgebra})]{linear span} \( \lin_R A \) of the subset \( A \) of \( M \) as the as the intersection of \( R \)-sub-semimodules of \( M \) containing \( A \). It coincides with the first-order submodel of \( M \) \hyperref[def:fol_generated_substructure]{generated} by \( A \).

    As shown in \cref{thm:span_via_linear_combinations}, the span can be characterized via \hyperref[def:linear_combination]{linear combinations}.

    \thmitem{def:semimodule/bisemimodule}\mcite[149]{Golan1999Semirings} An \( (R, S) \)-\term{bisemimodule} is a triple \( (M, R, S) \), where \( M \) is a commutative monoid that is both a left \( R \)-semimodule and a right \( S \)-semimodule, and the following associativity condition holds for all \( m \) in \( M \), all \( r \) in \( R \) and all \( t \) in \( T \):
    \begin{equation}\label{eq:def:semimodule/bisemimodule/associativity}
      (r \cdot_R m) \cdot_S t = r \cdot_R (m \cdot_S t).
    \end{equation}
  \end{thmenum}
\end{definition}
\begin{defproof}
  We will show that the two definitions of left \( R \)-semimodule are equivalent. For right \( R \)-semimodules, we can adapt the proof straightforwardly.

  \ImplicationSubProof{def:semimodule/action}{def:semimodule/operation} Fix a semiring homomorphism \( \varphi: R \to \End(M) \) and define the operation \( r \cdot x \coloneqq \varphi(r)(x) \).

  We will verify that all conditions from \cref{def:semimodule/operation} hold for this operation.

  \begin{itemize}
    \item By definition, \( \varphi \) is a monoid action of \( (R, \cdot) \) on \( (M, \bincirc) \); hence, \eqref{eq:def:semimodule/operation/action/identity} and \eqref{eq:def:semimodule/operation/action/compatibility} hold.

    \item The scalar distributivity condition \eqref{eq:def:semimodule/operation/scalar_addition_distributivity} holds because \( \varphi: R \to \End(M) \) is a semiring homomorphism:
    \begin{equation*}
      (r +_R s) \cdot x
      =
      \varphi(r +_R s)(x)
      =
      [\varphi(r) +_{\End(M)} \varphi(s)](x)
      =
      \varphi(r)(x) +_M \varphi(s)(x)
      =
      r \cdot x +_M s \cdot x.
    \end{equation*}

    \item Similarly, the zero scalar absorption condition \eqref{eq:def:semimodule/operation/absorption/scalar} holds:
    \begin{equation*}
      0_R \cdot x
      =
      \varphi(0_R)(x)
      =
      0_{\End(M)}(x)
      =
      0_M.
    \end{equation*}

    \item The vector distributivity condition \eqref{eq:def:semimodule/operation/vector_addition_distributivity} holds because, for every scalar \( r \) in \( R \), \( \varphi(r): M \to M \) is a monoid homomorphism:
    \begin{equation*}
      r \cdot (x +_M y)
      =
      \varphi(r)(x +_M y)
      =
      \varphi(r)(x) +_M \varphi(r)(y)
      =
      r \cdot x +_M r \cdot y
    \end{equation*}

    \item Similarly, the zero vector absorption condition \eqref{eq:def:semimodule/operation/absorption/vector} holds:
    \begin{equation*}
      r \cdot 0_M = \varphi(r)(0_M) = 0_M.
    \end{equation*}
  \end{itemize}

  \ImplicationSubProof{def:semimodule/operation}{def:semimodule/action} Let \( {\cdot}: R \times M \to M \) be an operation satisfying all conditions from \cref{def:semimodule/operation}.

  Consider the function \( \varphi: R \to \fun(M, M) \) sending \( r \) to the map \( x \mapsto r \cdot x \).

  \SubProof*{Proof that \( \varphi(r): M \to M \) is a monoid homomorphism} We have
  \begin{equation*}
    \varphi(r)(0_M)
    =
    r \cdot 0_M
    \reloset {\eqref{eq:def:semimodule/operation/absorption/vector}} =
    0_M
  \end{equation*}
  and
  \begin{equation*}
    \varphi(r)(x +_M y)
    =
    r \cdot (x +_M y)
    \reloset {\eqref{eq:def:semimodule/operation/vector_addition_distributivity}} =
    (r \cdot x) +_M (r \cdot y)
    =
    \varphi(r)(x) +_M \varphi(r)(y).
  \end{equation*}

  \SubProof*{Proof that \( \varphi: R \to \End(M) \) is a semiring homomorphism}

  The operation preserves both units because
  \begin{equation*}
    \varphi(0_R)
    =
    (x \mapsto 0_R \cdot x)
    \reloset {\eqref{eq:def:semimodule/operation/absorption/scalar}} =
    (x \mapsto 0_M)
    =
    0_{\End(M)}
  \end{equation*}
  and
  \begin{equation*}
    \varphi(1_R)
    =
    (x \mapsto 1_R \cdot x)
    \reloset {\eqref{eq:def:semimodule/operation/action/identity}} =
    (x \mapsto x)
    =
    \id_M.
  \end{equation*}

  We must also show that it preserves both binary operations. Clearly
  \begin{equation*}
    \varphi(r +_R s)
    =
    (x \mapsto (r +_R s) \cdot x)
    \reloset {\eqref{eq:def:semimodule/operation/scalar_addition_distributivity}} =
    (x \mapsto r \cdot x +_M s \cdot x)
    =
    \varphi(r) +_M \varphi(s).
  \end{equation*}

  For multiplication, we have
  \begin{equation*}
    \varphi(r \cdot_R s)
    =
    (x \mapsto (r \cdot_R s) \cdot x)
    \reloset {\eqref{eq:def:semimodule/operation/action/compatibility}} =
    (x \mapsto r \cdot (s \cdot x))
    =
    \parens[\big]{ x \mapsto \varphi(r)\parens[\big]{ \varphi(s)(x) } }
    =
    \varphi(r) \bincirc \varphi(s).
  \end{equation*}
\end{defproof}

\begin{definition}\label{def:homogeneous_function}\mimprovised
  We say that a function \( f: M \to N \) between \( R \)-\hyperref[def:semimodule]{(semi)modules} is \( R \)-\term[ru=однородная (\cite[416]{Зорич2019АнализЧасть1}), en=homogeneous (\cite[def. 2.3.1]{HillePhillips1996FunctionalAnalysis})]{homogeneous} of degree \( d \) if, for every \( r \) from \( R \) and every \( x \) from \( M \), we have
  \begin{equation}\label{eq:def:homogeneous_function}
    f(rx) = r^d \cdot f(x).
  \end{equation}

  We will shorten \enquote{homogeneous of degree \( 1 \)} to simply \enquote{homogeneous}.
\end{definition}
\begin{comments}
  \item We generalize real-valued homogeneous functions from \cref{def:real_homogeneous_function}.
\end{comments}

\begin{remark}\label{rem:two_sorted_semimodules}
  In \cref{def:semimodule/theory}, we presented a one-sorted \hyperref[def:fol_equational_theory]{equational} \hyperref[def:fol_theory]{first-order theory} for semimodules. For modules, this approach is used in \cite[\S 3.3.3]{TentZiegler2012ModelTheory}.

  The main advantage to this formulation is the ability to use the constructs from \fullref{sec:first_order_structures}, \fullref{sec:first_order_theories} and \fullref{sec:equational_theories} developed specifically for one-sorted first-order logic. The main disadvantage is the inability to formalize the theory since, breaking the convention from \cref{rem:language_alphabet_cardinality}, we must allow infinite signatures.

  An alternative is to use \hyperref[def:many_sorted_fol]{two-sorted first-order logic}. For modules, this approach is used in \cite[def. 8.3.4]{TroelstraVanDalen1988ConstructivismVol2} and \cite[78]{КолмогоровДрагалин2006Логика}.

  Let \( \BbbK \) be the sort of scalars and \( \BbbV \) --- sort of vectors. Then we can formulate the theory of semimodules via the axioms:
  \begin{align*}
    \qforall {\synx^{\syn\BbbV}} (\syn1_\BbbK \synmult \synx &\syneq \synx), \\
    \qforall {\synr^{\syn\BbbK}} \qforall {\syns^{\syn\BbbK}} \qforall {\synx^{\syn\BbbV}} ((\synr \synmult_\BbbK \syns) \synmult \synx &\syneq \synr \synmult (\syns \synmult \synx)), \\
    \qforall {\synr^{\syn\BbbK}} \qforall {\syns^{\syn\BbbK}} \qforall {\synx^{\syn\BbbV}} ((\synr \synplus_\BbbK \syns) \synmult \synx &\syneq (\synr \synmult \synx) \synplus_\BbbV (\syns \synmult \synx)), \\
    \qforall {\synr^{\syn\BbbK}} \qforall {\synx^{\syn\BbbV}} \qforall {\syny^{\syn\BbbV}} (\synr \synmult (\synx \synplus_\BbbK \syny) &\syneq (\synr \synmult \synx) \synplus_\BbbV (\synr \synmult \syny)), \\
    \qforall {\synx^{\syn\BbbV}} (\syn0_\BbbK \synmult \synx &\syneq \syn0_\BbbV), \\
    \qforall {\synr^{\syn\BbbK}} (\synr \synmult \syn0_\BbbV &\syneq \syn0_\BbbV).
  \end{align*}

  With this formalization, we gain the ability to study \hyperref[def:hol_homomorphism]{higher-order homomorphisms} such as \( \varphi = (\varphi_\BbbK, \varphi_\BbbV) \), where \( \varphi_\BbbK: R \to S \) is a semiring homomorphism and \( \varphi_\BbbV: M \to N \) is a function from the \( R \)-semimodule \( M \) to the \( S \)-semimodule \( N \) such that
  \begin{equation*}
    \varphi_\BbbV(r \cdot x) = \varphi_\BbbK(r) \cdot \varphi_\BbbV(x).
  \end{equation*}

  As noted in \cite{MathSE:many_sorted_fol_theory_for_vector_spaces}, however, in practice we consider \( R \)-semimodules over a \emph{fixed} semiring \( R \), so we will not gain much from this level of generality.
\end{remark}

\begin{proposition}\label{thm:def:semimodule}
  \( R \)-\hyperref[def:semimodule]{semimodules} have the following basic properties:
  \begin{thmenum}
    \thmitem{thm:def:semimodule/union} The union of a \hyperref[def:order_function/preserving]{ascending sequence} \( N_1 \subseteq N_2 \subseteq \cdots \) of \( R \)-\hyperref[def:semimodule/subobject]{sub-semimodules} of \( M \) is also an \( R \)-sub-semimodule of \( M \).
  \end{thmenum}
\end{proposition}
\begin{proof}
  \SubProofOf{thm:def:semimodule/union} Trivial.
\end{proof}

\begin{proposition}\label{thm:bisemimodule_over_submonoid}
  If \( S \) is (isomorphic to) a sub-semiring of \( R \), then \( R \) is an \( S \)-bisemimodule with scalar multiplication given by multiplication in \( R \).
\end{proposition}
\begin{comments}
  \item This result specializes to algebras over semirings in \cref{thm:algebra_over_subring}.
\end{comments}
\begin{proof}
  We must show that \( \cdot \) satisfied the conditions in \cref{def:semimodule/operation}.
  \begin{itemize}
    \item The identity law \eqref{eq:def:semimodule/operation/action/identity} holds because \( 1 \) is the multiplicative unit of \( R \).
    \item The associativity-like law \eqref{eq:def:semimodule/operation/action/compatibility} follows from associativity of multiplication.
    \item The two distributivity laws \eqref{eq:def:semimodule/operation/scalar_addition_distributivity} and \eqref{eq:def:semimodule/operation/vector_addition_distributivity} follow from left and right distributivity on \( R \).
    \item The absorption laws \eqref{eq:def:semimodule/operation/absorption/scalar} and \eqref{eq:def:semimodule/operation/absorption/vector} follow from absorption on semirings.
  \end{itemize}

  All the above also hold for right semimodules rather than left.
\end{proof}

\begin{proposition}\label{thm:commutative_monoid_is_semimodule}
  The \hyperref[def:category]{categories} \hyperref[def:monoid/category]{of commutative monoids} and \hyperref[def:semimodule/category]{of \( \BbbN \)-semimodules} are \hyperref[rem:category_similarity/isomorphism]{isomorphic}.

  More concretely, every commutative monoid \( M \) is an \( \BbbN \)-left semimodule with scalar multiplication given by the recursively defined multiplication operation \eqref{eq:def:additive_semigroup/iterated}.

  Conversely, in every \( \BbbN \)-semimodule, scalar multiplication matches the recursively defined multiplication.
\end{proposition}
\begin{comments}
  \item This result specializes to algebras over semirings in \cref{thm:semiring_is_natural_number_algebra} and modules over rings in \cref{thm:abelian_group_is_module}.
\end{comments}
\begin{proof}
  \SufficiencySubProof Let \( M \) be a commutative monoid. The operation \( \cdot: \BbbN \times M \to M \) defined in \cref{thm:semiring_characteristic_homomorphism} satisfies the conditions in \cref{def:semimodule/operation} as either a direct consequence of the definition or as a consequence of \cref{thm:monoid_distributivity}.

  The homomorphisms are thus also compatible.

  \NecessitySubProof Let \( M \) be a semimodule over \( \BbbN \). We will use induction to show that \eqref{eq:def:semimodule/operation/absorption/scalar} holds.
  \begin{itemize}
    \item For \( n = 0 \), this follows from the scalar absorption law \eqref{eq:def:semimodule/operation/absorption/scalar}.
    \item If \( n \cdot x = n \cdot x + x \), then by scalar distributivity, \( (n + 1) \cdot x = n \cdot x + 1 \cdot x \). The multiplicative identity law \eqref{eq:def:semimodule/operation/action/identity} then shows that \( 1 \cdot x = x \), which concludes our proof.
  \end{itemize}

  The homomorphisms are thus also compatible.
\end{proof}

\begin{example}\label{ex:def:semimodule}
  We list examples of \hyperref[def:semimodule]{semimodules}:
  \begin{thmenum}
    \thmitem{ex:def:semimodule/natural_number_span} The \hyperref[def:rational_numbers]{rational numbers} \( \BbbQ \) form a \hyperref[def:semimodule]{semimodule} over both the \hyperref[def:integers]{integers} \( \BbbZ \) and the \hyperref[def:natural_numbers]{natural numbers} \( \BbbN \).

    The \hyperref[def:semimodule/generated]{linear span} \( \lin_\BbbN \set{ 1 / 2 } \) consists of all \emph{positive} multiples of \( 1 / 2 \), so it is not a \( \BbbZ \)-module.

    It is thus an \( \BbbN \)-\hyperref[def:semimodule/subobject]{sub-semimodule} of \( \BbbQ \), but not a \( \BbbZ \)-\hyperref[def:semimodule/subobject]{sub-semimodule}.

    \thmitem{ex:def:semimodule/complex_conjugate} The \hyperref[def:complex_conjugation]{complex conjugation} map \( z \mapsto \oline z \) is \( \BbbR \)-linear because
    \begin{equation*}
      \oline{a(c + di)} = a(c - di) = a\oline{(c + di)},
    \end{equation*}
    but is not \( \BbbC \)-linear because
    \begin{equation*}
      \oline{(a + bi)(c + di)} = \oline{(ac - bd) + (bc + ad)i} = (ac - bd) - (bc + ad)i
    \end{equation*}
    and
    \begin{equation*}
      (a + bi)\oline{(c + di)} =(a + bi)(c - di) = (ac + bd) - (bc + ad)i.
    \end{equation*}
  \end{thmenum}
\end{example}

\paragraph{Semimodule direct sums}

\begin{definition}\label{def:function_support}\mcite[19]{Golan1999Semirings}
  We define the \term[bg=носител (\cite[58]{Боянов2008ЧислениМетоди}), ru=носитель (\cite[135]{КанторовичАкилов1984ФункциональныйАнализ})]{support} of a function \( f: S \to R \) from any set \( S \) to a semiring \( R \) as the set
  \begin{equation*}
    \supp(f) \coloneqq \set{ x \in S \given f(x) \neq 0_R }.
  \end{equation*}
\end{definition}
\begin{comments}
  \item The definition is mostly useful in analysis, but it also naturally fits in \hyperref[def:truth_value_algebra]{truth value algebras}, where, as discussed in \cref{rem:boolean_valued_functions_and_relations}, \( \supp(f) \) is the underlying relation of a Boolean-valued function.
\end{comments}

\begin{definition}\label{def:semimodule_direct_sum}\mimprovised
  We define the \term{external direct sum} or simply \term{direct sum} \( \bigoplus_{k \in \mscrK} M_k \) of the family of \( R \)-\hyperref[def:semimodule]{semimodules} \( \seq{ M_k }_{k \in \mscrK} \) as the subset of their \hyperref[def:fol_direct_product]{direct product} \( \prod_{k \in \mscrK} M_k \) consisting of tuples with finite \hyperref[def:function_support]{support}.

  The sums and scalar products of tuples with finite support also have finite support, hence the direct sum is an \( R \)-\hyperref[def:semimodule/subobject]{sub-semimodule} of the direct product.

  \begin{thmenum}
    \thmitem{def:semimodule_direct_sum/inclusion} For every \( m \in \mscrK \), we define the following \term{canonical inclusion} \hyperref[def:semimodule/morphism]{homomorphism}:
    \begin{equation*}
      \begin{aligned}
        &\iota_m: M_m \to \bigoplus_{k \in \mscrK} M_k, \\
        &\iota_m(x) \coloneqq \begin{rcases}
          \begin{cases}
            x, &k = m \\
            0, &k \neq m
          \end{cases}
        \end{rcases}_{k \in \mscrK}
      \end{aligned}
    \end{equation*}

    \thmitem{def:semimodule_direct_sum/power} In case all summands are equal to \( M \), we denote the direct sum by \( M^{\oplus \mscrK} \).
    \thmitem{def:semimodule_direct_sum/internal} If all summands are submodels of \( M \) and if the sum \( \bigoplus_{k \in \mscrK} M_k \) is isomorphic to \( M \), we call it an \term{internal direct sum} and treat the tuple \( \seq{ x_k }_{k \in \mscrK} \) as the product \( x_{k_1} x_{k_2} \cdots x_{k_n} \) of the elements of \( M \) distinct from zero.
  \end{thmenum}
\end{definition}

\begin{proposition}\label{thm:semimodule_coproduct}
  The \hyperref[def:discrete_category_limits]{categorical coproduct} of the family \( \seq{ M_k }_{k \in \mscrK} \) in the \hyperref[def:semimodule/category]{category of \emph{commutative} semimodules} is their \hyperref[def:semimodule_direct_sum]{direct sum} \( \bigoplus_{k \in \mscrK} M_k \).
\end{proposition}
\begin{proof}
  First note that the sum \( A \coloneqq \bigoplus_{k \in \mscrK} A_k \) with the \hyperref[def:semimodule_direct_sum/inclusion]{inclusions} \( \iota \coloneqq \seq{ \iota_k }_{k \in \mscrK} \) are a \hyperref[def:category_of_cones/cocone]{cocone} for the \hyperref[def:discrete_category]{discrete} \hyperref[def:categorical_diagram]{diagram} \( \seq{ A_k }_{k \in \mscrK} \).

  Let \( (C, \alpha) \) also be a cocone. We want to define a semimodule homomorphism
  \begin{equation*}
    l_C: \bigoplus_{k \in \mscrK} A_k \to A
  \end{equation*}
  such that, for every \( m \in \mscrK \) and \( x \in M_m \),
  \begin{equation*}
    \alpha_m(x) = l_A(\iota_m(x)).
  \end{equation*}

  Thus, the value of \( l_A(c) \) on members of the inclusion \( \iota_m[M_m] \) is entirely determined by \( \alpha_m \). This suggests the definition
  \begin{equation*}
    l_C(\seq{ x_k }_{k \in \mscrK}) \coloneqq \prod_{k \in \mscrK}^n \alpha_k(x_k).
  \end{equation*}

  We discuss well-definedness of infinitary operations in direct sums in \cref{rem:binary_operations_to_infinitary/direct_sum}.
\end{proof}

\paragraph{Free semimodules}

\begin{definition}\label{def:free_semimodule}\mimprovised
  Fix a \hyperref[def:semiring]{semiring} \( R \). We associate with every \hyperref[def:set]{plain set} \( A \) its \term{free \( R \)-semimodule} \( R^{\oplus A} \) over \( R \) defined as the set
  \begin{equation*}
    R^{\oplus A} \coloneqq \bigoplus_{x \in A} R = \set{ t: A \to R \given t \T{has finite \hyperref[def:function_support]{support}} }.
  \end{equation*}

  \begin{thmenum}
    \thmitem{def:free_semimodule/operations} \Cref{thm:fol_equational_theory_models/functions} implies that \( R^{\oplus A} \) is a semiring with the elementwise addition and multiplication inherited from \( R \). Scalar multiplication by \( r \in R \) can be defined as multiplication by the \hyperref[def:constant_function]{constant function} at \( r \).

    \thmitem{def:free_semimodule/inclusion} We start with the canonical inclusion
    \begin{equation*}
      \begin{aligned}
        &\iota_A: A \to R^{\oplus A}, \\
        &\iota_A(x) \coloneqq \parens[\Bigg]
          {
            y \mapsto \begin{rcases}
              \begin{cases}
                1_R, &y = x \\
                0_R, &y \neq x
              \end{cases}
            \end{rcases}
          }
      \end{aligned}
    \end{equation*}
  \end{thmenum}
\end{definition}
\begin{comments}
  \item In accordance with \cref{rem:free_construction_univalence}, we refer to any (semi)module isomorphic to \( R^{\oplus A} \) as \enquote{free}, and to \( R^{\oplus A} \) itself as \enquote{the} free (semi)module. For \hyperref[def:module]{modules} this reduces to finding a \hyperref[def:hamel_basis]{Hamel basis} --- see \cref{rem:free_module}.

  \item We will use the notation \cref{rem:free_semimodule_notation} for semimodules.

  \item In the case when \( R \) is the semiring \( \BbbN \) of natural numbers, \( \BbbN^{\oplus A} \) is the set of finite \hyperref[def:multiset]{multisets} over \( S \).
\end{comments}

\begin{remark}\label{rem:free_semimodule_notation}
  We can use the inclusion \cref{def:free_semimodule/inclusion} of \( A \) into its \hyperref[def:free_semimodule]{free semimodule} \( R^{\oplus A} \) to introduce a simpler notation.

  For any \( t: A \to R^{\oplus A} \) and any \( a \in A \), denote by \( t_a \) the constant function taking the value \( t(a) \)\fnote{In \cref{def:free_semimodule/operations} we have defined scalar multiplication by \( t(a) \) as multiplication with \( t_a \). Having a clear distinction between functions and scalars makes the derivation of \eqref{eq:def:free_semimodule/sum} clearer}.

  Then
  \begin{equation*}
    t_a(x) \cdot \iota_A(a)(x)
    =
    \begin{cases}
      t(a) \cdot_R 0_R, &a \neq x, \\
      t(a) \cdot_R 1_R, &a = x
    \end{cases}
  \end{equation*}

  On other words, the product \( t_a(x) \cdot \iota_A(a)(x) \) is zero for every \( x \neq a \) and coincides with \( t(x) \) when \( x = a \). As discussed in \cref{rem:binary_operations_to_infinitary/direct_sum}, an infinite sum is well-defined if only finitely many summands are nonzero, thus
  \begin{equation*}
    t(x) = \sum_{a \in A} t_a(x) \cdot_R \iota_A(a)(x),
  \end{equation*}
  which can also be written more succinctly as
  \begin{equation*}
    t = \sum_{a \in A} t_a \cdot \iota_A(a).
  \end{equation*}

  To make the notation simpler, we can write \( a \) instead of \( \iota_A(a) \):
  \begin{equation}\label{eq:def:free_semimodule/sum}
    t = \sum_{a \in A} t_a \cdot a.
  \end{equation}

  Outside this remark, we will regard \( t_a \) as a scalar and \( a \) as a vector.
\end{remark}

\begin{theorem}[Free semimodule universal property]\label{thm:free_semimodule_universal_property}
  Fix a semiring \( R \) and a set \( A \). The \hyperref[def:free_semimodule]{free \( R \)-semimodule} \( R^{\oplus A} \) over \( R \) is the unique up to a unique isomorphism semimodule that satisfies the following \hyperref[rem:universal_mapping_property]{universal mapping property}:
  \begin{displayquote}
    For every semimodule \( M \) over \( R \) and every function \( e: A \to M \), there exists a unique \( R \)-semimodule homomorphism \( \Phi_e: R^{\oplus A} \to M \) such that the following diagram commutes:
    \begin{equation}\label{eq:thm:free_semimodule_universal_property/diagram}
      \begin{aligned}
        \includegraphics[page=1]{output/thm__free_semimodule_universal_property}
      \end{aligned}
    \end{equation}
  \end{displayquote}
\end{theorem}
\begin{comments}
  \item Via \cref{rem:universal_mapping_property}, \( A \mapsto R^{\oplus A} \) becomes \hyperref[def:category_adjunction]{left adjoint} to the \hyperref[def:concrete_category]{forgetful functor}
  \begin{equation*}
    U: \cat{SMod}_R \to \cat{Set}.
  \end{equation*}

  \item \hyperref[con:free_construction/evaluation]{Evaluation homomorphisms} for polynomials are defined via a map based on \( \Phi_e \). We thus refer to \( \Phi_e \) as an \enquote{evaluation homomorphism}.
\end{comments}
\begin{proof}
  \ExistenceSubProof For every function \( e: A \to M \), we want
  \begin{equation*}
    \Phi_e(\iota(x)) = e(x).
  \end{equation*}

  This suggests the definition
  \begin{equation*}
    \Phi_e(\sum_{a \in A} t_a \cdot a) \coloneqq \sum_{x \in A} \sum_{a \in A} t_a \cdot e(a).
  \end{equation*}

  \UniquenessSubProof Fix a linear map \( \Psi_e: R^{\oplus A} \to M \) such that
  \begin{equation*}
    \Psi_e(\iota(x)) = e(x).
  \end{equation*}

  We will use induction on the number of nonzero entries in \( \seq{ t_x }_{x \in A} \) to show that \( \Phi_e \) and \( \Psi_e \) coincide. The base case where all entries are zero is obvious since
  \begin{equation*}
    \Psi_e(\iota(0_R)) = e(0_r) = \Phi_e(\iota(0_R)).
  \end{equation*}

  Now suppose that
  \begin{equation*}
    \Psi_e(\iota(x)) = e(x),
  \end{equation*}
  and that \( \Psi_e \) coincides with \( \Phi_e \) for every element of \( R^{\oplus A} \) with \( n - 1 \) entries. Given \( \seq{ t_x }_{x \in A} \) with \( n \) nonzero entries, fix a nonzero \( t_{x_0} \) and define
  \begin{equation}
    t'_x \coloneqq \begin{cases}
      0,   &x = x_0, \\
      t_x, &\T{otherwise.}
    \end{cases}
  \end{equation}

  Then
  \begin{balign*}
    \Psi_e(\seq{ t_x }_{x \in A})
    &=
    \sum_{x \neq x_0} t_x \cdot e(x) + t_{x_0} \cdot e(x_0)
    = \\ &=
    \Psi_e(\seq{ t'_x }_{x \in A}) + t_{x_0} \cdot e(x_0)
    \reloset {\T{ind.}} = \\ &=
    \Phi_e(\seq{ t'_x }_{x \in A})
    = \\ &=
    \Phi_e(\seq{ t_x }_{x \in A}).
  \end{balign*}

  This concludes the proof.
\end{proof}

\begin{definition}\label{def:linear_combination}\mimprovised
  Fix an \hyperref[def:indexed_family]{indexed} \hyperref[def:formal_language/alphabet]{alphabet} \( \seq{ X_k }_{k \in \mscrK} \). These will act as indeterminates in the sense of \cref{con:free_construction/indeterminate}.

  A \term[ru=линейная комбинация (\cite[\S 3.2]{Тыртышников2007ЛинейнаяАлгебра}), en=linear combination (\cite[153]{Golan1999Semirings})]{linear combination} in these indeterminates is simply an element of the \hyperref[def:free_semimodule]{free semimodule} over them.

  For every linear combination \( t \), we can adapt the notation from \cref{rem:free_semimodule_notation} to write
  \begin{equation}\label{eq:def:linear_combination}
    \sum_{k \in \mscrK} t_k \cdot X_k.
  \end{equation}

  We refer to the scalars \( t_k \) in \eqref{eq:def:linear_combination} as \term[ru=коэффициенты (линейной комбинации) (\cite[\S 3.2]{Тыртышников2007ЛинейнаяАлгебра}), en=coefficients (\cite[39]{FriedbergInselSpence2018LinearAlgebra})]{coefficients} and to the products \( t_k \cdot X_k \) as \term{terms}. If \( t_k = 0_R \), we may, by \enquote{abuse of language}, say that \term{no term} corresponds to \( X_k \).

  \begin{thmenum}
    \thmitem{def:linear_combination/trivial} If all coefficients are zero, we say that the linear combination is \term[ru=тривиальная (линейная комбинация) (\cite[\S 3.3]{Тыртышников2007ЛинейнаяАлгебра}), en=trivial (linear combination) (\cite[8]{Treil2017LinearAlgebraDoneWrong})]{trivial}.

    \thmitem{def:linear_combination/sum} For every \( R \)-semimodule \( M \) and every family \( \seq{ m_k }_{k \in \mscrK} \) of vectors in \( M \), \fullref{thm:free_semimodule_universal_property} allows us to evaluate \eqref{eq:def:linear_combination} to obtain a vector in \( M \):
    \begin{equation}\label{eq:def:linear_combination/sum}
      \sum_{k \in \mscrK}^n t_k \cdot m_k.
    \end{equation}

    Only finitely many terms are nonzero, so this sum is well-defined even when \( \mscrK \) is infinite. We discuss this more generally in \cref{rem:binary_operations_to_infinitary}.

    We will refer to \eqref{eq:def:linear_combination/sum} as a \enquote{linear combination \emph{of the vectors}} \( \seq{ m_k }_{k \in \mscrK} \).
  \end{thmenum}
\end{definition}
\begin{comments}
  \item Linear combinations and their sums are thus an instance of the syntax-semantics duality described in \cref{con:syntax_semantics_duality}. We base our presentation on \hyperref[def:polynomial_algebra]{polynomials}, where this distinction is made clear.

  Unfortunately, it is not established practice to distinguish between a linear combination and its sum, which formally renders the discussions of coefficients and terms nonsensical. Linear combinations are \hyperref[con:assumed_knowledge]{assumed knowledge} and even informal definitions can rarely be found. Even those authors like \textcite[\S 3.2]{Тыртышников2007ЛинейнаяАлгебра} and \textcite[6]{Treil2017LinearAlgebraDoneWrong} that give definitions do not delve into the syntax-semantics duality.
\end{comments}

\begin{proposition}\label{thm:span_via_linear_combinations}
  For a set \( A \) in an \( R \)-\hyperref[def:semimodule]{semimodule} \( M \), the \hyperref[def:semimodule/generated]{linear span} of \( A \) equals the set of all \hyperref[def:linear_combination]{linear combinations} over \( A \).
\end{proposition}
\begin{comments}
  \item Compare this result to the analogous result \cref{thm:generators_via_polynomials} for algebras and polynomials.
\end{comments}
\begin{proof}
  \Cref{fig:thm:span_via_linear_combinations} shows an \hyperref[con:abstract_syntax_tree]{abstract syntax tree} for a given linear combination, which can be traversed and evaluated to obtain a vector in \( M \). This vector must be a member of the span of \( S \) since the latter is closed under vector addition and scalar multiplication with members of \( S \). Hence, the set \( L \) of all linear combinations over \( S \) is a subset of the span.

  Generalizing the syntax tree construction from \cref{fig:thm:span_via_linear_combinations}, we see that \( L \) satisfies \cref{thm:fol_generated_substructure_iteration}, and is thus a submodule of \( M \). Since the span is the smallest module containing \( S \), we have \( L = \lin S \).

\end{proof}

\paragraph{Free commutative monoids}

\begin{remark}\label{rem:free_commutative_monoid_as_quotient}
  Consider the \hyperref[def:free_monoid]{free monoid} \( A^* \) over some \hyperref[def:set]{plain set} \( A \) and also the \hyperref[def:free_semimodule]{free \( \BbbN \)-semimodule} \( \BbbN^{\oplus A} \).

  Define the homomorphism
  \begin{equation*}
    \begin{aligned}
      &\varphi: A^* \to \BbbN^{\oplus A}, \\
      &\varphi(x_1 \cdots x_n) \coloneqq \seq*{ \underbrace{\sum_{k=1}^n 1_{a = x_k}}_{\mathclap{\T*{repetitions of} a \T*{among} x_1 \cdots x_n} }}_{a \in A}
    \end{aligned}
  \end{equation*}

  By \cref{thm:fol_homomorphism_induces_congruence}, \( \varphi \) induces a \hyperref[def:fol_congruence]{monoid congruence} \( \cong \) on \( A^* \) where two strings are congruent if they have the same amount of each symbol from \( A \).

  This ensures that the strings \enquote{\( abc \)}, \enquote{\( bac \)}, \enquote{\( bca \)}, \enquote{\( cba \)}, \enquote{\( cab \)} and \enquote{\( acb \)} are congruent --- these are precisely all variations of \enquote{\( abc \)} that can be obtained via \hyperref[def:transposition]{transpositions} due to commutativity.

  The \hyperref[def:fol_quotient_structure]{quotient} \( A^* / \cong \) is isomorphic to \( \BbbN^{\oplus A} \) as a monoid. This motivates the definition of free commutative monoids in \cref{def:free_commutative_monoid}.
\end{remark}

\begin{definition}\label{def:free_commutative_monoid}\mimprovised
  We associate with every \hyperref[def:set]{plain set} \( A \) its \term{free commutative monoid} defined as the \hyperref[def:free_semimodule]{free \( \BbbN \)-module} \( \BbbN^{\oplus A} \).
\end{definition}
\begin{comments}
  \item In accordance with \cref{rem:free_construction_univalence}, we refer to any monoid isomorphic to \( \BbbN^{\oplus A} \) as \enquote{free}, and to \( \BbbN^{\oplus A} \) itself as \enquote{the} free monoid.

  \item We regard \( \BbbN^{\oplus A} \) only as a monoid, without generally considering scalars.

  \item The relation to \hyperref[def:free_monoid]{free monoids} is given in \cref{rem:free_commutative_monoid_as_quotient}.

  \item See \cref{ex:def:fol_free_term_model/free_commutative_monoid} for a comparison of \( \BbbN^{\oplus A} \) to the isomorphic \hyperref[def:fol_free_term_model]{free term model} of the \hyperref[def:monoid/commutative]{first-order theory of commutative monoids}.
\end{comments}
