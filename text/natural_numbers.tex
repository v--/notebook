\section{Natural numbers}\label{sec:natural_numbers}

\begin{definition}\label{def:natural_numbers}
  We define the set of \term[bg=естествени числа (\cite[371]{ГеновМиховскиМоллов1991Алгебра}), ru=натуральные числа (\cite[def. 11.1]{Проскуряков1951ТеоретическиеОсновыАрифметики})]{natural numbers} \( \BbbN \) as a fixed \hyperref[def:peano_arithmetic_standard_model]{standard model} of \hyperref[def:peano_arithmetic]{Peano arithmetic}.

  The choice of model does not matter most of the time, and we can choose any from \cref{thm:peano_arithmetic_standard_models}. Yet, the \hyperref[def:peano_arithmetic_term_model]{standard term model} \( \mscrN \) is more useful when dealing with formal arithmetic, while the \hyperref[thm:smallest_inductive_set_existence]{smallest inductive set} \( \omega \) is more useful across \fullref{ch:set_theory}.

  For convenience, we use the familiar \hyperref[def:positional_number_system/decimal]{decimal notation}
  \begin{align*}
    0 &\coloneqq \syn0,                   \\
    1 &\coloneqq \synS \syn0,             \\
    2 &\coloneqq \synS \synS \syn0,       \\
    3 &\coloneqq \synS \synS \synS \syn0, \\
      &\vdots                             \\
  \end{align*}
\end{definition}
\begin{comments}
  \item In accordance with \cref{rem:mathematical_logic_conventions/models_as_sets}, unless we are working with explicit logical formulas, we will generally reason about \( \BbbN \) in the ambient metatheory and treat it, its members and its operations as sets in \hyperref[def:axiom_of_universes]{\logic{ZFC+U}}.
\end{comments}

\paragraph{Algebraic structure of \( \BbbN \)}

\begin{proposition}\label{thm:natural_number_addition_properties}
  The \hyperref[def:natural_numbers]{natural numbers} \( \BbbN \) with addition, as characterized by \hyperref[def:peano_arithmetic]{Peano arithmetic}, form a \hyperref[def:binary_operation/cancellative]{cancellative} \hyperref[def:binary_operation/commutative]{commutative} \hyperref[def:zerosumfree]{zerosumfree} \hyperref[def:monoid]{monoid} with \( 0 \) as the neutral element.
\end{proposition}
\begin{comments}
  \item We generalize this to \hyperref[def:ordinal]{ordinal numbers} in \cref{thm:ordinal_addition_algebraic_properties} and to \hyperref[def:cardinal]{cardinal numbers} in \cref{thm:cardinal_addition_algebraic_properties}.
\end{comments}
\begin{proof}
  \SubProofOf[def:binary_operation/associative]{associativity} Fix natural numbers \( n \), \( m \) and \( k \). We will prove associativity by induction on \( k \), simultaneously for all \( n \) and \( m \).
  \begin{itemize}
    \item If \( k = 0 \), we have
    \begin{equation*}
      (n + m) + 0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
      n + m
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
      n + (m + 0).
    \end{equation*}

    \item If the inductive hypothesis holds for \( k \), then
    \begin{balign*}
      (n + m) + S(k)
      &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} =
      S((n + m) + k)
      \reloset {\T{ind.}} = \\ &=
      S(n + (m + k))
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} = \\ &=
      n + S(m + k)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} = \\ &=
      n + m + S(k).
    \end{balign*}
  \end{itemize}

  \SubProofOf[def:binary_operation/commutative]{commutativity} We prove \( n + m = m + n \) using induction on \( m \), simultaneously for all \( n \). We will need an additional base case for \( m = 1 \).

  \begin{itemize}
    \item If \( m = 0 \), we use nested induction on \( n \) with a fixed \( m \).
    \begin{itemize}
      \item If \( n = m = 0 \), clearly \( n + m = 0 + 0 = m + n \).
      \item If the inductive hypothesis holds for \( n \), then
      \begin{balign*}
        S(n) + 0
        &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
        S(n)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} = \\ &=
        S(n + 0)
        \reloset {\T{ind.}} = \\ &=
        S(0 + n)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} = \\ &=
        0 + S(n).
      \end{balign*}
    \end{itemize}

    \item If \( m = 1 \), we use nested induction on \( n \) with a fixed \( m \).
    \begin{itemize}
      \item If \( n = 0 \), then
      \begin{equation*}
        0 + 1
        \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} =
        S(0)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
        S(0) + 0.
      \end{equation*}

      \item If the inductive hypothesis holds for \( n \), then
      \begin{balign*}
        S(n) + 1
        &\reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} =
        (n + 1) + 1
        \reloset {\T{ind.}} = \\ &=
        (1 + n) + 1
        \reloset {\eqref{eq:def:binary_operation/associative}} = \\ &=
        1 + (n + 1)
        \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} = \\ &=
        1 + S(n).
      \end{balign*}
    \end{itemize}

    \item If the inductive hypothesis holds for \( m \), we again use nested induction on \( n \) with a fixed \( m \).
    \begin{itemize}
      \item If \( n = 0 \), then
      \begin{equation*}
        0 + S(m)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} =
        S(0 + m)
        \reloset {\T{ind.}} =
        S(m + 0)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
        S(m)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
        S(m) + 0.
      \end{equation*}

      \item Finally, if the inductive hypothesis holds for \( n \), then
      \begin{balign*}
        S(n) + S(m)
        &\reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} =
        (n + 1) + (m + 1)
        \reloset {\T{ind.}} = \\ &=
        m + (n + 1) + 1
        = \\ &=
        m + (1 + n) + 1
        \reloset {\eqref{eq:def:binary_operation/associative}} = \\ &=
        (m + 1) + (n + 1)
        \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} = \\ &=
        S(m) + S(n).
      \end{balign*}
    \end{itemize}
  \end{itemize}

  \SubProofOf[def:monoid]{neutral element} We have \( n + 0 = n \) by \eqref{eq:def:peano_arithmetic/theory/PA4} and \( 0 + n = n \) by commutativity.

  \SubProofOf[def:binary_operation/cancellative]{cancellation} Let \( n + k = m + k \). We will prove that \( n = m \) by induction on \( k \), simultaneously for all \( n \) and \( m \).
  \begin{itemize}
    \item For \( k = 0 \), this follows from \eqref{eq:def:peano_arithmetic/theory/PA4}.
    \item If the inductive hypothesis holds for \( k \), then
    \begin{equation*}
      n + S(k)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} =
      S(n + k)
      \reloset {\T{ind.}} =
      S(m + k)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA5}} =
      m + S(k)
    \end{equation*}
  \end{itemize}

  \SubProof{Proof that \( \BbbN \) is \hyperref[def:zerosumfree]{zerosumfree}} We must show that \( n + m = 0 \) implies \( n = m = 0 \). We will use induction on \( m \), simultaneously for all \( n \).

  \begin{itemize}
    \item If \( m = 0 \), then \( n + m = n \) by \eqref{eq:def:peano_arithmetic/theory/PA4}, and hence \( n = 0 \).

    \item If the inductive hypothesis holds for \( m \), then \( n + S(m) = S(n + m) \) by \eqref{eq:def:peano_arithmetic/theory/PA5}, and \( 0 = S(n + m) \) contradicts \eqref{eq:def:peano_arithmetic/theory/PA1}.
  \end{itemize}
\end{proof}

\begin{lemma}\label{thm:natural_number_multiplication_and_monoid_exponentiation}
  \hyperref[def:natural_numbers]{Natural number} multiplication, as characterized by \hyperref[def:peano_arithmetic]{Peano arithmetic}, coincides with \hyperref[con:additive_semigroup/multiplication]{multiplication in commutative monoids}, as defined in \cref{def:monoid/exponentiation} for a natural number and a monoid member.
\end{lemma}
\begin{proof}
  Denote Peano multiplication by \( \cdot \) and monoid multiplication by \( \times \).

  We will show that \( n \cdot m = n \times m \) by induction on \( m \), simultaneously for all \( n \).
  \begin{itemize}
    \item For the base case \( m = 0 \), \eqref{eq:def:peano_arithmetic/theory/PA6} implies that \( n \cdot 0 = 0 \), while \( n \times 0 \) is by definition the neutral element \( 0 \).

    \item If the inductive hypothesis holds for \( m \), then \eqref{eq:def:peano_arithmetic/theory/PA7} implies that
    \begin{equation*}
      n \cdot S(m) = n \cdot m + n,
    \end{equation*}
    which is precisely the definition of \( n \times (m + 1) \). \Cref{thm:peano_arithmetic_successor_and_addition} implies that \( S(m) = m + 1 \), which concludes the proof.
  \end{itemize}
\end{proof}

\begin{proposition}\label{thm:natural_number_multiplication_properties}
  The \hyperref[def:natural_numbers]{natural numbers} \( \BbbN \) with multiplication, as characterized by \hyperref[def:peano_arithmetic]{Peano arithmetic}, form a \hyperref[def:binary_operation/commutative]{commutative} \hyperref[def:monoid]{monoid} with \( 1 \) as the neutral element.

  When combined with addition, the natural numbers become an \hyperref[def:entire_semiring]{entire} \hyperref[def:semiring/commutative]{commutative semiring}.
\end{proposition}
\begin{comments}
  \item We generalize this to \hyperref[def:ordinal]{ordinal numbers} in \cref{thm:ordinal_multiplication_algebraic_properties} and to \hyperref[def:cardinal]{cardinal numbers} in \cref{thm:cardinal_multiplication_algebraic_properties}.
\end{comments}
\begin{proof}
  \SubProof{Proof that \( 1 \) is a \hyperref[def:monoid]{neutral element}} Multiplication by \( 1 \) on the right preserves any natural number:
  \begin{equation*}
     n \cdot 1
     \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
     n \cdot 0 + n
     \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
     0 + n
     \reloset {\ref{thm:natural_number_addition_properties}} =
     n.
  \end{equation*}

  Multiplication from the left is handled by induction on \( n \).
  \begin{itemize}
    \item If \( n = 0 \), then
    \begin{equation*}
       0 \cdot 1
       \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
       0.
    \end{equation*}

    \item If the inductive hypothesis holds for \( n \), then
    \begin{equation*}
       1 \cdot S(n)
       \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
       1 \cdot n + 1
       \reloset {\T{ind.}} =
       n + 1
       \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} =
       S(n).
    \end{equation*}
  \end{itemize}

  \SubProofOf[def:semiring/left_distributivity]{distributivity} We will prove that \( (n + m) \cdot k = n \cdot k + n \cdot k \) by induction on \( k \), simultaneously for all \( n \) and \( m \).

  \begin{itemize}
    \item If \( k = 0 \),
    \begin{equation*}
      (n + m) \cdot 0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
      0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
      0 + 0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
      n \cdot 0 + m \cdot 0.
    \end{equation*}

    \item If the inductive hypothesis holds for \( k \), then
    \begin{balign*}
      (n + m) \cdot S(k)
      &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
      (n + m) + (n + m) \cdot k
      \reloset {\T{ind.}} = \\ &=
      (n + m) + n \cdot k + n \cdot k
      \reloset {\ref{thm:natural_number_addition_properties}} = \\ &=
      (n + n \cdot k) + (m + m \cdot k)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} = \\ &=
      n \cdot k + m \cdot k.
    \end{balign*}
  \end{itemize}

  \SubProofOf[def:binary_operation/associative]{associativity} With distributivity proven, we will show \( (n \cdot m) \cdot k = n \cdot (m \cdot k) \) by induction on \( k \), simultaneously for all \( n \) and \( m \).
  \begin{itemize}
    \item If \( k = 0 \), then
    \begin{equation*}
      (n \cdot m) \cdot 0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
      0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
      n \cdot 0
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
      n \cdot (m \cdot 0)
    \end{equation*}

    \item If the inductive hypothesis holds for \( k \), then
    \begin{balign*}
      (n \cdot m) \cdot S(k)
      &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
      n \cdot m + (n \cdot m) \cdot k
      \reloset {\T{ind.}} = \\ &=
      n \cdot m + n \cdot (m \cdot k)
      \reloset {\eqref{eq:def:semiring/left_distributivity}} = \\ &=
      n \cdot (m + m \cdot k)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} = \\ &=
      n \cdot (m \cdot S(k)).
    \end{balign*}
  \end{itemize}

  \SubProofOf[def:binary_operation/commutative]{commutativity} We will show \( n \cdot m = m \cdot n \) by induction on \( m \), simultaneously for all \( n \).
  \begin{itemize}
    \item If \( m = 0 \), we use nested induction on \( n \) for a fixed \( m \).
    \begin{itemize}
      \item If \( n = 0 \), trivially
      \begin{equation*}
        n \cdot m
        =
        0 \cdot 0
        =
        m \cdot n.
      \end{equation*}

      \item If the inductive hypothesis holds for \( n \), then
      \begin{equation*}
        S(n) \cdot 0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
        0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} =
        0 + 0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} =
        0 \cdot n + 0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
        0 \cdot S(n).
      \end{equation*}
    \end{itemize}

    \item If the inductive hypothesis holds for \( m \), we again use nested induction on \( n \) with a fixed \( m \).
    \begin{itemize}
      \item If \( n = 0 \), then
      \begin{balign*}
        0 \cdot S(m)
        &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
        0 \cdot m + 0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA4}} = \\ &=
        0 \cdot m
        \reloset {\T{ind.}} = \\ &=
        m \cdot 0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} = \\ &=
        0
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA6}} = \\ &=
        S(m) \cdot 0.
      \end{balign*}

      \item Finally, if the inductive hypothesis holds for \( n \), then
      \begin{balign*}
        S(n) \cdot S(m)
        &\reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
        S(n) \cdot m + S(n)
        \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} = \\ &=
        (n + 1) \cdot m + S(n)
        \reloset {\eqref{eq:def:semiring/left_distributivity}} = \\ &=
        n \cdot m + 1 \cdot m + S(n)
        \reloset {\eqref{eq:def:monoid/theory/neutral}} = \\ &=
        n \cdot m + m + S(n)
        \reloset {\T{ind.}} = \\ &=
        m \cdot n + m + S(n)
        \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} = \\ &=
        m \cdot S(n) + S(n)
        \reloset {\eqref{eq:def:monoid/theory/neutral}} = \\ &=
        m \cdot S(n) + 1 \cdot S(n)
        \reloset {\eqref{eq:def:semiring/left_distributivity}} = \\ &=
        (m + 1) \cdot S(n)
        \reloset {\eqref{eq:thm:peano_arithmetic_successor_and_addition}} = \\ &=
        S(m) \cdot S(n).
      \end{balign*}
    \end{itemize}
  \end{itemize}

  \SubProofOf[def:semiring/absorption]{absorption} From \eqref{eq:def:peano_arithmetic/theory/PA6} it follows that \( n0 = 0 \) for every \( n \), and from commutativity of multiplication it also follows that \( 0n = 0 \).

  \SubProof{Proof of no zero divisors} We will now show that \( \BbbN \) has no zero divisors.

  Namely, we will show that \( n \cdot m = 0 \) implies \( n = m = 0 \) by induction on \( m \), simultaneously for all \( n \).
  \begin{itemize}
    \item If \( m = 0 \), \eqref{eq:def:peano_arithmetic/theory/PA6} implies that \( n = 0 \).
    \item If the inductive hypothesis holds for \( m \), then
    \begin{equation*}
      n \cdot S(m)
      \reloset {\eqref{eq:def:peano_arithmetic/theory/PA7}} =
      S(n \cdot m),
    \end{equation*}
    which cannot be zero due to \eqref{eq:def:peano_arithmetic/theory/PA1}.
  \end{itemize}
\end{proof}

\paragraph{Order structure of \( \BbbN \)}

\begin{proposition}\label{thm:natural_numbers_are_well_ordered}
  The \hyperref[def:natural_numbers]{natural numbers} are \hyperref[def:well_ordered_set]{well-ordered} by the relation \( \leq \) defined by \cref{def:peano_arithmetic_ordering}.

  Furthermore, \( \BbbN \) is an \hyperref[def:ordered_semiring]{ordered semiring}, that is, the nonstrict order \( \leq \) is compatible with addition and multiplication.
\end{proposition}
\begin{proof}
  \SubProofOf[def:binary_relation/reflexive]{reflexivity} We have \( n \leq n \) since \( n + 0 = n \).

  \SubProofOf[def:binary_relation/transitive]{transitivity} If \( n \leq m \) and \( m \leq k \), then there exist numbers \( a \) and \( b \) such that \( m = n + a \) and \( k = m + b \).

  Then \( k = n + (a + b) \), hence \( n \leq k \).

  \SubProofOf[def:binary_relation/connected]{connectedness} We will show that \( n \leq m \) or \( m \leq n \) using induction on \( m \), simultaneously on all \( n \).
  \begin{itemize}
    \item If \( m = 0 \), then \( m \leq n \) because \( 0 + n = n \).
    \item If the inductive hypothesis holds for \( m \), we have two cases
    \begin{itemize}
      \item If \( n \leq m \), then \( m = n + a \) for some \( a \) and hence \( S(m) = n + S(a) \); i.e. \( n \leq S(m) \).
      \item If \( m \leq n \), then \( n = m + b \) for some \( b \), and we have two cases to consider, based on whether \( b \) has a predecessor (see \cref{ex:peano_arithmetic_predecessor_existence}).
      \begin{itemize}
        \item If \( b = 0 \), then \( m = n \), hence \( S(m) = n + 1 \) and \( n \leq S(m) \).
        \item If \( b = S(c) \) for some \( c \), then \( n = m + S(c) = S(m + c) = S(m) + c \), hence \( S(m) \leq n \).
      \end{itemize}
    \end{itemize}
  \end{itemize}

  \SubProofOf[def:chain_condition]{descending chain condition} We will use induction on \( n \) to show that every descending chain ending with \( n \) stabilizes.

  \begin{itemize}
    \item For the base case \( n = 0 \), consider the chain
    \begin{equation*}
      \cdots \leq m_2 \leq m_1 \leq 0.
    \end{equation*}

    By transitivity, \( m_i \leq 0 \) for every index \( i \). \Cref{thm:peano_arithmetic_order_compatibility} implies that \( m_i = 0 \) or \( m_i < 0 \). But the latter option contradicts \cref{thm:peano_arithmetic_less_than_zero}.

    It follows that \( m_i = 0 \) for every index \( i \).

    \item Suppose that the inductive hypothesis holds for \( n \), and consider the chain
    \begin{equation*}
      \cdots \leq m_2 \leq m_1 \leq S(n).
    \end{equation*}

    By transitivity, \( m_i \leq S(n) \) for every index \( i \). Then either \( m_i = S(n) \) for every \( i \), in which case the chain is constant, or \( m_i \leq n \), in which case we can use the inductive hypothesis.
  \end{itemize}

  \SubProofOf[def:ordered_semigroup]{compatibility with addition} We must show that \( n + k \leq m + k \) if \( n \leq m \).

  This turns out to be quite simple: we have \( m = n + a \) for some number \( a \), so
  \begin{equation*}
    m + k = (n + a) + k = (n + k) + a.
  \end{equation*}

  \SubProofOf[def:ordered_semiring]{compatibility with multiplication} We must show that \( n \geq 0 \) and \( m \geq 0 \) together imply \( n \cdot m \geq 0 \).

  This holds vacuously because, as shown in \cref{thm:peano_arithmetic_less_than_zero}, all natural numbers are nonnegative.
\end{proof}

\begin{proposition}\label{thm:natural_number_multiplicative_invertibility}
  The only \hyperref[def:divisibility/invertible]{invertible} \hyperref[def:natural_numbers]{natural number} is \( 1 \).
\end{proposition}
\begin{proof}
  Suppose that \( nm = 1 \).
  \begin{itemize}
    \item If either \( m = 0 \) or \( n = 0 \), then \( nm = 0 \) due to absorption of \( 0 \), contradicting the assumption \( nm = 1 \).
    \item Otherwise \cref{ex:peano_arithmetic_predecessor_existence} implies that \( m \) has a predecessor, say \( k \), so
    \begin{equation*}
      1 = nm = n S(k) = nk + n.
    \end{equation*}

    \begin{itemize}
      \item If \( k = 0 \), then \( n = 1 \). In this case \( m = S(0) = 1 \).
      \item Otherwise, \( 1 = nm > n \geq 1 \), which is again a contradiction.
    \end{itemize}
  \end{itemize}

  Only in the case \( n = m = 1 \) we have not reached a contradiction.
\end{proof}

\begin{proposition}\label{thm:natural_numbers_archimedean}
  The semiring of natural numbers is \hyperref[def:archimedean_semiring]{Archimedean}. In fact, since there are no negative natural numbers, we can state the property as follows:
  \begin{displayquote}
    For every pair of natural numbers \( m \) and \( n \), if \( m \) is nonzero, there exists a nonzero natural number \( k \) such that \( n < km \).
  \end{displayquote}
\end{proposition}
\begin{proof}
  We will use induction on \( n \).

  \begin{itemize}
    \item If \( n = 0 \), trivially \( n = 0 < m \) (with \( k = 1 \)).

    \item Suppose that the inductive hypothesis holds for \( n \) and consider \( S(n) \). There exists some nonzero \( k \) such that \( n < km \).

    Then there exists some number \( a \) such that \( km = n + 1 = S(n) \). Clearly
    \begin{equation*}
      (k + 1)m = km + k = S(n) + k,
    \end{equation*}
    so \( S(n) < (k + 1) m \).
  \end{itemize}
\end{proof}

\begin{lemma}\label{thm:bounded_set_of_natural_numbers_finite}
  There are only finitely many \hyperref[def:natural_numbers]{natural numbers} strictly less than a given natural number.
\end{lemma}
\begin{proof}
  We will use induction on \( n \) to show the initial segment \( \BbbN_{<n} \) is finite.
  \begin{itemize}
    \item It follows from \cref{thm:peano_arithmetic_admissible_rules/less_than_zero} that the set \( \BbbN_{<0} \) is empty.
    \item Suppose that the inductive hypothesis holds for \( n \).

    \Cref{thm:peano_arithmetic_admissible_rules/less_successor_to_leq} implies that \( m < S(n) \) if either \( m < n \), in which case \( m \) belongs to \( \BbbN_{<n} \), or otherwise \( m = n \).

    Therefore,
    \begin{equation*}
      \BbbN_{<S(n)} = \BbbN_{<S(n)} \cup \set{ n }
    \end{equation*}

    Since \( \BbbN_{<n} \) is finite, so is \( \BbbN_{<S(n)} \).
  \end{itemize}
\end{proof}

\begin{proposition}\label{thm:natural_numbers_confinal_equivalence}
  For a set of \hyperref[def:natural_numbers]{natural numbers}, the following are equivalent:
  \begin{thmenum}
    \thmitem{thm:natural_numbers_confinal_equivalence/cofinal} It is \hyperref[def:cofinal_set]{cofinal}.
    \thmitem{thm:natural_numbers_confinal_equivalence/unbounded} It is \hyperref[def:extremal_points/bounds]{unbounded from above}.
    \thmitem{thm:natural_numbers_confinal_equivalence/infinite} It is \hyperref[def:set_finiteness]{infinite}.
  \end{thmenum}
\end{proposition}
\begin{proof}
  Let \( A \) be a set of natural numbers.

  \EquivalenceSubProof{thm:natural_numbers_confinal_equivalence/cofinal}{thm:natural_numbers_confinal_equivalence/unbounded} \Cref{thm:def:totally_ordered_set/cofinal_iff_unbounded} implies that \( A \) is cofinal if and only if it is unbounded from above.

  \ImplicationSubProof{thm:natural_numbers_confinal_equivalence/unbounded}{thm:natural_numbers_confinal_equivalence/infinite} If \( A \)is unbounded from above, we can recursively build a strictly increasing sequence from its elements. This sequence has infinitely many elements, hence so does \( A \).

  \ImplicationSubProof{thm:natural_numbers_confinal_equivalence/infinite}{thm:natural_numbers_confinal_equivalence/unbounded} If \( A \) is infinite, by (the contraposition of) \cref{thm:bounded_set_of_natural_numbers_finite}, it must be unbounded from above.
\end{proof}
